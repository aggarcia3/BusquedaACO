003package esei.si.alejandrogg.busquedaACO;
004
005import java.util.Collections;
006import java.util.HashMap;
007import java.util.Map;
008import java.util.HashSet;
009import java.util.Set;
010import java.util.regex.Pattern;
011import java.text.Normalizer;
012import java.util.concurrent.atomic.LongAdder;
013
014import jason.asSyntax.ASSyntax;
015import jason.asSyntax.Atom;
016import jason.asSyntax.Literal;
017
018/**
019 * Esta clase es una fabricación pura, cuyas responsabilidades versan sobre el mantenimiento
020 * de las percepciones de los agentes en coordinación con el entorno, {@link Mundo}.
021 *
022 * Esta clase no debe de invocar métodos en su instancia de {@link Mundo} asociada que puedan
023 * requerir de información no inicializada. Para más información acerca de qué está inicializado
024 * y qué no cuando se crea este gestor, véase la implementación de {@link Mundo}.
025 *
026 * @see Mundo
027 * @author Alejandro González García
028 */
029public final class GestorPercepciones {
030    /**
031     * El autómata que reconoce la expresión regular que siguen los caracteres a sustituir
032     * por otro caracter.
033     */
034    private static final Pattern regexSustituciones = Pattern.compile("^[0-9]|\\s");
035    /**
036     * El autómata que reconoce la expresión regular de los símbolos que no forman parte de
037     * un átomo válido.
038     */
039    private static final Pattern regexCaracteresProhibidos = Pattern.compile("[^a-zA-Z0-9_]");
040    /**
041     * El literal que unifica con el usado para notificar a los agentes del paso del tiempo en el sistema.
042     */
043    private static final Literal literalPasoTiempoUnif = ASSyntax.createLiteral("pasoTiempo", ASSyntax.createVar());
044    /**
045     * El literal que unifica con el usado para notificar a los agentes de la llegada de un nuevo ciclo.
046     */
047    private static final Literal literalSiguienteCicloUnif = ASSyntax.createLiteral("siguienteCiclo", ASSyntax.createVar());
048
049    /**
050     * La instancia del entorno asociada con este gestor.
051     */
052    private final Mundo mundo;
053    /**
054     * Los átomos que representan en AgentSpeak de manera inequívoca a cada localidad.
055     * Es seguro que varios hilos operen sobre él de manera concurrente.
056     */
057    private final Map<Localidad, Atom> atomosLocalidades;
058    /**
059     * Las localidades representadas por átomos en AgentSpeak, de manera inequívoca.
060     * Es seguro que varios hilos operen sobre él de manera concurrente.
061     */
062    private final Map<String, Localidad> localidadesAtomos;
063    /**
064     * El objeto cuyo monitor se usará para realizar exclusión mutua en las actualizaciones
065     * de percepciones de feromonas (carreteras), para garantizar que los agentes siempre
066     * ven un entorno consistente.
067     */
068    private final Object candadoActualizacionFeromona = new Object();
069    /**
070     * Se utiliza para asignar a las percepciones que sean necesarias un identificador lo más unívoco
071     * posible. Esto sirve para que Jason pueda diferenciar diferentes percepciones de paso de tiempo,
072     * y los agentes no vuelvan a reincorporar esa percepción tras borrarla de su BC, por alguna razón
073     * que, para variar, no está documentada explícitamente.
074     */
075    private LongAdder idPercepcion = new LongAdder();
076
077    /**
078     * Crea un nuevo gestor de percepciones, asociado con un entorno (o mundo) dado, al
079     * que le añadirá las percepciones iniciales relacionadas con el grafo de carreteras
080     * en uso.
081     * @param mundo El mundo a asociar.
082     * @throws IllegalArgumentException Si el mundo a asociar es nulo.
083     */
084    GestorPercepciones(final Mundo mundo) {
085        if (mundo == null) {
086            throw new IllegalArgumentException("Un gestor de percepciones no puede tener un mundo asociado nulo.");
087        }
088        this.mundo = mundo;
089
090        final Set<Localidad> localidades = mundo.getGrafoCarreteras().localidades();
091        final Set<String> funtoresUsados = new HashSet<>((int) (localidades.size() / 0.75) + 1, 0.75f);
092
093        // De paso que guardamos los átomos asociados a cada localidad en memoria, para no tener que computarlos
094        // frecuentemente, también comprobamos que los funtores asociados sean unívocos, para garantizar el
095        // correcto funcionamiento del sistema. Y, naturalmente, también añadimos percepciones para el mapeo
096        // átomo-nombre
097        final Map<Localidad, Atom> atomosLocalidades = new HashMap<>((int) (localidades.size() / 0.75) + 1, 0.75f);
098        final Map<String, Localidad> localidadesAtomos = new HashMap<>((int) (localidades.size() / 0.75) + 1, 0.75f);
099        for (final Localidad l : localidades) {
100            final Atom at = calcularAtomoLocalidad(l);
101            final String func = at.getFunctor();
102
103            if (funtoresUsados.contains(func)) {
104                throw new IllegalArgumentException("No se pudo generar un átomo unívoco para identificar la localidad " + l.getNombre());
105            }
106
107            mundo.addPercept(
108                ASSyntax.createLiteral("localidadANombre",
109                    at, ASSyntax.createString(l.getNombre())
110                )
111            );
112
113            atomosLocalidades.put(l, at);
114            localidadesAtomos.put(func, l);
115            funtoresUsados.add(func);
116        }
117        // Como no necesitaremos actualizar los contenidos de los mapas, hacerlos de solo
118        // lectura es una manera de bajo coste de garantizar la seguridad con varios hilos
119        // de ejecución, aún bajo condiciones imprevistas
120        this.atomosLocalidades = Collections.unmodifiableMap(atomosLocalidades);
121        this.localidadesAtomos = Collections.unmodifiableMap(localidadesAtomos);
122
123        // Crear percepciones del tipo:
124        // carretera(A, B, Distancia, IntFeromona)
125        for (final Carretera c : mundo.getGrafoCarreteras().carreteras()) {
126            final Localidad[] nodosCarretera = c.getLocalidades();
127            mundo.addPercept(
128                ASSyntax.createLiteral("carretera",
129                    atomosLocalidades.get(nodosCarretera[0]), atomosLocalidades.get(nodosCarretera[1]),
130                    ASSyntax.createNumber(c.getDistancia()), ASSyntax.createNumber(Mundo.FEROMONA_INICIAL)
131                )
132            );
133        }
134
135        // Crear percepciones del tipo:
136        // localidadInicio(localidad)
137        // localidadDestino(localidad)
138        mundo.addPercept(
139            ASSyntax.createLiteral("localidadInicio",
140                atomosLocalidades.get(mundo.getLocalidadInicio())
141            )
142        );
143        mundo.addPercept(
144            ASSyntax.createLiteral("localidadDestino",
145                atomosLocalidades.get(mundo.getLocalidadDestino())
146            )
147        );
148
149        // Crear percepciones del tipo:
150        // importanciaFeromona(Alfa)
151        // importanciaDistancia(Beta)
152        mundo.addPercept(
153            ASSyntax.createLiteral("importanciaFeromona",
154                ASSyntax.createNumber(mundo.getAlfa())
155            )
156        );
157        mundo.addPercept(
158            ASSyntax.createLiteral("importanciaDistancia",
159                ASSyntax.createNumber(mundo.getBeta())
160            )
161        );
162
163        // Aunque no están documentadas las implicaciones de llamar a este método
164        // de Jason por ninguna parte, un examen del código fuente indica que invoca
165        // el método wakeUpSense() de los agentes en la arquitectura centralizada,
166        // que tiene algo que ver con el control de la ejecución de los ciclos de percepción.
167        // Así pues, invocar este método supongo que puede interpretarse como una indicación
168        // a los agentes de que deberían de refrescar sus percepciones del entorno más pronto
169        // que tarde (algo que tiene sentido por su nombre, pero quería asegurarme), aunque
170        // normalmente los agentes ya ejecutan esos ciclos por sí solos
171        mundo.informAgsEnvironmentChanged();
172    }
173
174    /**
175     * Reenvía a todos los agentes la percepción de paso de tiempo al siguiente
176     * instante de tiempo discreto. Este método no está diseñado para ser ejecutado
177     * por varios hilos de ejecución en paralelo.
178     */
179    void percibirPasoTiempo() {
180        mundo.removePerceptsByUnif(literalPasoTiempoUnif);
181        mundo.addPercept(ASSyntax.createLiteral("pasoTiempo", ASSyntax.createNumber(idPercepcion.intValue())));
182
183        mundo.informAgsEnvironmentChanged();
184
185        idPercepcion.increment(); // Este incremento no es atómico respecto a la consulta del valor, pero no nos importa
186    }
187
188    /**
189     * Informa a todos los agentes del nuevo nivel de feromona que hay en una carretera determinada.
190     * Este método es seguro para ser ejecutado por varios hilos concurrentemente.
191     * @param carretera La carretera cuyo nivel de feromona ha variado.
192     * @param nuevoNivel El nuevo nivel de feromona de la carretera. El paso por parámetro de este
193     * valor implica evitar tener que volver a consultárselo a la carretera, y ayuda a reducir contención
194     * de hilos.
195     */
196    void actualizarNivelFeromona(final Carretera carretera, final double nuevoNivel) {
197        final Localidad[] nodosCarretera = carretera.getLocalidades();
198        final double distancia = carretera.getDistancia();
199
200        mundo.removePerceptsByUnif(
201            ASSyntax.createLiteral("carretera",
202                atomosLocalidades.get(nodosCarretera[0]), atomosLocalidades.get(nodosCarretera[1]),
203                ASSyntax.createNumber(distancia), ASSyntax.createVar()
204            )
205        );
206        mundo.addPercept(
207            ASSyntax.createLiteral("carretera",
208                atomosLocalidades.get(nodosCarretera[0]), atomosLocalidades.get(nodosCarretera[1]),
209                ASSyntax.createNumber(distancia), ASSyntax.createNumber(nuevoNivel)
210            )
211        );
212
213        mundo.informAgsEnvironmentChanged();
214    }
215
216    /**
217     * Envía a todos los agentes la percepción de que ha llegado el siguiente ciclo. Este método no está
218     * diseñado para ser ejecutado por varios hilos de ejecución en paralelo.
219     */
220    void siguienteCiclo() {
221        mundo.removePerceptsByUnif(literalSiguienteCicloUnif);
222        mundo.addPercept(ASSyntax.createLiteral("siguienteCiclo", ASSyntax.createNumber(idPercepcion.intValue())));
223
224        mundo.informAgsEnvironmentChanged();
225
226        idPercepcion.increment(); // Este incremento no es atómico respecto a la consulta del valor, pero no nos importa
227    }
228
229    /**
230     * Obtiene la localidad correspondiente al átomo de AgentSpeak dado.
231     * @param atomo El átomo cuya localidad asociada obtener.
232     * @return La localidad identificada por el átomo. De no corresponderse
233     * el átomo con ninguna localidad, el valor de retorno puede ser nulo.
234     */
235    public Localidad localidadAtomo(final Atom atomo) {
236        return localidadesAtomos.get(atomo == null ? null : atomo.getFunctor());
237    }
238
239    /**
240     * Obtiene el átomo identificativo de una localidad, usado en las percepciones asociadas.
241     * Esta operación normaliza el nombre de la localidad de manera que pueda ser interpretado
242     * de manera segura como un átomo. Este método no garantiza que el resultado sea unívoco; es
243     * decir, que para dos localidades diferentes el resultado siempre sea diferente.
244     * @param l La localidad de la que obtener su átomo.
245     * @return El devandicho átomo.
246     */
247    private static Atom calcularAtomoLocalidad(final Localidad l) {
248        // Descomponer caracteres Unicode compuestos en secuencias de combinación,
249        // eliminando distinciones semánticas inútiles para comparaciones binarias
250        // (esta operación separa, por ejemplo, las tildes de las vocales)
251        String nombreAtomo = Normalizer.normalize(l.getNombre(), Normalizer.Form.NFKD);
252
253        // Sustituir espacios en blanco y dígito inicial por guiones bajos
254        nombreAtomo = regexSustituciones.matcher(nombreAtomo).replaceAll("_");
255        // Eliminar caracteres Unicode "extraños"
256        nombreAtomo = regexCaracteresProhibidos.matcher(nombreAtomo).replaceAll("");
257
258        // Finalmente, convertir a minúsculas
259        nombreAtomo = nombreAtomo.toLowerCase();
260
261        return ASSyntax.createAtom(nombreAtomo);
262    }
