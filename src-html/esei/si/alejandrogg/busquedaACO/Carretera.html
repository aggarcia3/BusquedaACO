003package esei.si.alejandrogg.busquedaACO;
004
005import java.util.concurrent.atomic.AtomicReference;
006import java.util.function.UnaryOperator;
007
008/**
009 * Modela una carretera, que se trata de un nombre más semántico para un arco
010 * etiquetado y no dirigido de un grafo, sobre el que hormigas pueden depositar
011 * feromonas (que se puede considerar como otro parámetro de la etiqueta).
012 * @author Alejandro González García
013 */
014public final class Carretera {
015    /**
016     * La localidad que está en un extremo de la carretera.
017     */
018    private final Localidad a;
019    /**
020     * La localidad que está en otro extremo de la carretera.
021     */
022    private final Localidad b;
023    /**
024     * La distancia entre ambas localidades implicadas (etiqueta del grafo).
025     */
026    private final double distancia;
027    /**
028     * La coordenada X (abscisa) del punto en la imagen que representa el grafo
029     * al que pertenece donde se mostrará información sobre la carretera.
030     * Tal imagen está asociada con su vista.
031     */
032    private final int infoX;
033    /**
034     * La coordenada Y (ordenada) del punto en la imagen que representa el grafo
035     * al que pertenece donde se mostrará información sobre la carretera.
036     * Tal imagen está asociada con su vista.
037     */
038    private final int infoY;
039    /**
040     * La cantidad de feromona depositada en esta carretera por las hormigas
041     * (agentes).
042     */
043    private final AtomicReference<Double> nivelFeromona = new AtomicReference<>(Mundo.FEROMONA_INICIAL);
044
045    /**
046     * Crea una nueva carretera.
047     * @param a La localidad que está en un extremo de la carretera.
048     * @param b La localidad que está en otro extremo de la carretera.
049     * @param distancia La distancia entre ambas localidades implicadas (etiqueta del grafo).
050     * @param infoX La coordenada X (abscisa) del punto donde se mostrará información de esta carretera.
051     * @param infoY La coordenada Y (ordenada) del punto donde se mostrará información de esta carretera.
052     * @throws NullPointerException Si alguno de los parámetros a o b son nulos.
053     * @throws IllegalArgumentException Si la distancia es menor que 0.
054     */
055    public Carretera(final Localidad a, final Localidad b, final double distancia, final int infoX, final int infoY) {
056        if (a == null || b == null) {
057            throw new NullPointerException("Las localidades que conforman una carretera no pueden ser nulas.");
058        }
059        if (distancia < 0) {
060            throw new IllegalArgumentException("La distancia no puede ser menor que 0.");
061        }
062        if (infoX < 0 || infoY < 0) {
063            throw new IllegalArgumentException("Alguna coordenada es negativa, cuando no puede serlo.");
064        }
065        this.a = a;
066        this.b = b;
067        this.distancia = distancia;
068        this.infoX = infoX;
069        this.infoY = infoY;
070    }
071
072    /**
073     * Obtiene las dos localidades por las que pasa esta carretera.
074     * @return Las devandichas localidades.
075     */
076    Localidad[] getLocalidades() {
077        return new Localidad[]{ a, b };
078    }
079
080    /**
081     * Obtiene la longitud de la propia carretera; es decir, la distancia para ir
082     * de una localidad a la otra usando esta carretera.
083     * @return La devandicha distancia.
084     */
085    double getDistancia() {
086        return distancia;
087    }
088
089    /**
090     * Obtiene la coordenada X (abscisa) del punto donde se mostrará información de esta carretera.
091     * @return La devandicha coordenada.
092     */
093    int getInfoX() {
094        return infoX;
095    }
096
097    /**
098     * Obtiene la coordenada Y (ordenada) del punto donde se mostrará información de esta carretera.
099     * @return La devandicha coordenada.
100     */
101    int getInfoY() {
102        return infoY;
103    }
104
105    /**
106     * Establece el nivel de feromona depositado actualmente en esta carretera, a partir
107     * de un cálculo realizado sobre el nivel de feromona anterior. Dicho cálculo se
108     * realiza de manera atómica: ningún otro hilo podrá ver resultados de la operación
109     * hasta que se complete, y se garantiza que otros hilos lo verán en cuanto se complete.
110     * @param gestorPercepciones El gestor de percepciones a usar para informar a las hormigas
111     * del cambio del nivel de feromona. Si es nulo, no se informará a las hormigas de ello.
112     * @param operacion La operación a realizar sobre el nivel de feromona anterior.
113     */
114    public void actualizarNivelFeromona(final GestorPercepciones gestorPercepciones, final UnaryOperator<Double> operacion) {
115        if (gestorPercepciones != null) {
116            gestorPercepciones.actualizarNivelFeromona(this, nivelFeromona.updateAndGet(operacion));
117        } else {
118            nivelFeromona.updateAndGet(operacion);
119        }
120    }
121
122    /**
123     * Obtiene el nivel de feromona depositada actualmente en esta carretera.
124     * @return El descrito dato.
125     */
126    double getNivelFeromona() {
127        return nivelFeromona.get();
128    }
129
130    /**
131     * {@inheritDoc} Dos carreteras son iguales si sus localidades inicio y destino son iguales,
132     * sin importar el orden.
133     */
134    @Override
135    public boolean equals(final Object otra) {
136        boolean toret = otra instanceof Carretera;
137
138        if (toret) {
139            final Carretera otraCarretera = (Carretera) otra;
140            toret = a.equals(otraCarretera.a) && b.equals(otraCarretera.b) || a.equals(otraCarretera.b) && b.equals(otraCarretera.a);
141        }
142
143        return toret;
144    }
145
146    @Override
147    public int hashCode() {
148        return a.hashCode() ^ b.hashCode();
149    }
150
151    @Override
152    public String toString() {
153        final StringBuilder sb = new StringBuilder(a.toString());
154
155        sb.append(", ");
156        sb.append(b);
157        sb.append(" (");
158        sb.append(distancia);
159        sb.append(")");
160
161        return sb.toString();
162    }
163}
