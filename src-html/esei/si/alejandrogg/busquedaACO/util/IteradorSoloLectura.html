003package esei.si.alejandrogg.busquedaACO.util;
004
005import java.util.Iterator;
006import java.util.function.Consumer;
007
008/**
009 * Implementa un {@link Iterator} sobre una colección que nunca soporta la eliminación
010 * de elementos. Esencialmente, este Iterator es un adaptador para otro Iterator.
011 * @author Alejandro González García
012 */
013public final class IteradorSoloLectura<E> implements Iterator<E> {
014    /**
015     * El Iterator del que este depende para ejecutar la lógica de implementación.
016     */
017    private final Iterator<E> iteradorInterno;
018
019    /**
020     * Crea una nueva instancia de esta clase, a partir de otro Iterator.
021     * @param iterador El iterador a usar, para ejecutar sobre él las operaciones de
022     * iteración.
023     * @throws IllegalArgumentException Si el parámetro es nulo.
024     */
025    public IteradorSoloLectura(final Iterator<E> iterador) {
026        if (iterador == null) {
027            throw new IllegalArgumentException("No es posible adaptar un iterador nulo a un iterador de solo lectura.");
028        }
029        this.iteradorInterno = iterador;
030    }
031
032    @Override
033    public void forEachRemaining(final Consumer<? super E> accion) {
034        iteradorInterno.forEachRemaining(accion);
035    }
036
037    @Override
038    public boolean hasNext() {
039        return iteradorInterno.hasNext();
040    }
041
042    @Override
043    public E next() {
044        return iteradorInterno.next();
045    }
046
047    @Override
048    public void remove() {
049        throw new UnsupportedOperationException("El iterador sobre esta colección es de solo lectura.");
050    }
051}
