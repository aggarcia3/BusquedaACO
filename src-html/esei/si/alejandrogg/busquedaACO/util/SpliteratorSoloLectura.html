003package esei.si.alejandrogg.busquedaACO.util;
004
005import java.util.Spliterator;
006import java.util.function.Consumer;
007
008/**
009 * Implementa un {@link Spliterator} inmutable y de tamaño definido sobre otro Spliterator,
010 * del que hereda otras características que tenga. Es, por tanto, un adaptador.
011 * @author Alejandro González García
012 */
013public final class SpliteratorSoloLectura<E> implements Spliterator<E> {
014    /**
015     * Las características que este Spliterator aporta al Spliterator usado internamente.
016     */
017    private static final int CARACTERISTICAS = Spliterator.IMMUTABLE | Spliterator.SIZED | Spliterator.SUBSIZED;
018
019    /**
020     * El Spliterator del que este depende para ejecutar la lógica de implementación.
021     */
022    private final Spliterator<E> spliteratorInterno;
023
024    /**
025     * Crea una nueva instancia de esta clase, cuyo Spliterator interno es el especificado.
026     * @param spliteratorInterno El Spliterator interno deseado.
027     * @throws IllegalArgumentException Si el Spliterator interno es nulo.
028     */
029    public SpliteratorSoloLectura(final Spliterator<E> spliteratorInterno) {
030        if (spliteratorInterno == null) {
031            throw new IllegalArgumentException("No se puede crear un Spliterator de solo lectura usando un Spliterator interno nulo.");
032        }
033        this.spliteratorInterno = spliteratorInterno;
034    }
035
036    @Override
037    public boolean tryAdvance(final Consumer<? super E> accion) {
038        return spliteratorInterno.tryAdvance(accion);
039    }
040
041    @Override
042    public void forEachRemaining(final Consumer<? super E> accion) {
043        spliteratorInterno.forEachRemaining(accion);
044    }
045
046    @Override
047    public Spliterator<E> trySplit() {
048        try {
049            return new SpliteratorSoloLectura<>(spliteratorInterno.trySplit());
050        } catch (IllegalArgumentException exc) {
051            return null;
052        }
053    }
054
055    @Override
056    public long estimateSize() {
057        return spliteratorInterno.estimateSize();
058    }
059
060    @Override
061    public int characteristics() {
062        return (spliteratorInterno.characteristics() | CARACTERISTICAS) & ~Spliterator.CONCURRENT;
063    }
064}
