028/**
029 * Implementa lógica de uso común por parte de algoritmos de búsqueda basados
030 * en optimización por colonias de hormigas. De esta forma, las subclases de esta clase pueden
031 * ahorrarse la reimplementación de tal lógica, y centrarse más en codificar lo que las
032 * diferencia.
033 *
034 * De hecho, esta clase abstracta contiene suficiente lógica como para constituir por ella misma
035 * una implementación (algo ineficiente y peculiar) de un algoritmo de búsqueda voraz estocástico, que no
036 * modifica los niveles de feromona del entorno. En consecuencia, lo más habitual es que las subclases se
037 * centren en añadirle estrategias concretas de actualización de feromonas.
038 *
039 * @author Alejandro González García
040 */
041abstract class BaseAlgoritmo implements Algoritmo {
042    /**
043     * El nombre identificativo de este algoritmo. Este nombre es proporcionado
044     * por la implementación concreta del algoritmo (subclase), no por el usuario.
045     */
046    private final String nombre;
047    /**
048     * El grafo de carreteras sobre el que se ejecuta el algoritmo.
049     */
050    protected final GrafoCarreteras grafoCarreteras;
051    /**
052     * El gestor de percepciones del entorno sobre el que se ejecuta el algoritmo.
053     */
054    protected final GestorPercepciones gestorPercepciones;
055    /**
056     * Los ciclos máximos que el algoritmo ejecutará.
057     */
058    private final int ciclosMaximos;
059    /**
060     * El ciclo actual en el que se encuentra la ejecución del algoritmo.
061     */
062    private final LongAdder ciclo;
063    /**
064     * La localidad de destino asociada a este algoritmo.
065     */
066    private final Localidad localidadDestino;
067
068    /**
069     * El conjunto de hormigas que se han desplazado mediante un arco en la iteración actual.
070     */
071    private final Set<String> hormigasRecorridoArco = Collections.newSetFromMap(new ConcurrentHashMap<>());
072    /**
073     * El conjunto de hormigas que han encontrado un camino en el ciclo actual, sea solución o no.
074     */
075    private final Set<String> hormigasCaminoEncontrado = Collections.newSetFromMap(new ConcurrentHashMap<>());
076    /**
077     * La cola de todos los caminos encontrados hasta el momento por las hormigas, sean
078     * solución o no, en el ciclo actual.
079     */
080    private final Queue<Camino> caminosEncontrados = new ConcurrentLinkedQueue<>();
081    /**
082     * Un mapa que relaciona las carreteras por las que pasa un camino encontrado en el ciclo actual
083     * que conduce a una solución.
084     */
085    private final Map<Carretera, List<Camino>> carreterasCaminoSolucion = new ConcurrentHashMap<>();
086    /**
087     * Objeto cuyo monitor se usa para establecer exclusión mutua y relaciones pasa-antes en los accesos
088     * a {@link mejorSolucionEncontrada}.
089     */
090    private final Object candadoMejorSolucionEncontrada = new Object();
091    /**
092     * El mejor camino solución encontrado hasta el momento por las hormigas, entre todos los ciclos.
093     */
094    private Camino mejorSolucionEncontrada = null;
095
096    /**
097     * La causa de terminación del algoritmo, si ha terminado todavía.
098     */
099    private volatile RazonTerminacion razonTerminacion = null;

101    /**
102     * Crea un nuevo algoritmo a partir de los parámetros especificados.
103     * @param grafoCarreteras El grafo de carreteras sobre el que operará este algoritmo.
104     * @param gestorPercepciones El gestor de percepciones, que proporciona funcionalidades
105     * para actualizar el estado del entorno en base a los resultados del algoritmo.
106     * @param ciclosMaximos Los ciclos máximos a ejecutar por el algoritmo.
107     * @param localidadDestino La localidad de destino, de la que el algoritmo intentará
108     * encontrar el camino más corto hacia ella.
109     * @param nombre El nombre identificativo de este algoritmo. Debe de ser proporcionado
110     * por la implementación de esta clase, no por el usuario.
111     * @throws IllegalArgumentException Si grafoCarreteras es nulo, gestorPercepciones es nulo,
112     * ciclosMaximos es menor que 1, localidadDestino es nula o nombre es nulo.
113     */
114    protected BaseAlgoritmo(final GrafoCarreteras grafoCarreteras, final GestorPercepciones gestorPercepciones, final int ciclosMaximos, final Localidad localidadDestino, final String nombre) {
115        if (grafoCarreteras == null) {
116            throw new IllegalArgumentException("No se puede crear un algoritmo asociado a un grafo de carreteras nulo.");
117        }
118        if (gestorPercepciones == null) {
119            throw new IllegalArgumentException("No se puede crear un algoritmo asociado a un gestor de percepciones nulo.");
120        }
121        if (ciclosMaximos < 1) {
122            throw new IllegalArgumentException("Los ciclos máximos a ejecutar por un algoritmo no pueden tener un valor menor que 1.");
123        }
124        if (localidadDestino == null) {
125            throw new IllegalArgumentException("No se puede crear un algoritmo para encontrar caminos a una localidad de destino nula.");
126        }
127        if (nombre == null) {
128            throw new IllegalArgumentException("El nombre identificativo asociado a un algoritmo no puede ser nulo.");
129        }
130        this.grafoCarreteras = grafoCarreteras;
131        this.gestorPercepciones = gestorPercepciones;
132        this.ciclosMaximos = ciclosMaximos;
133        this.localidadDestino = localidadDestino;
134        this.nombre = nombre;
135        this.ciclo = new LongAdder();
136        ciclo.increment(); // Comenzar número de ciclo en 1
137    }

139    /**
140     * Valida e interpreta una lista de localidades recibida de una hormiga, que define el camino que ha recorrido.
141     * Se asume que la lista de localidades es no nula, aunque sus elementos pueden serlo o no.
142     * @param listaLocalidades La lista de localidades a validar.
143     * @return El camino recorrido por la hormiga, generado a partir de lo que ha indicado en la lista, en un estado
144     * inmutable.
145     * @throws IllegalArgumentException Si la lista tiene menos de dos elementos, algún elemento de la
146     * lista no es un átomo, un átomo de la lista no representa una localidad, o no se ha podido encontrar
147     * una carretera que conecte a una localidad del camino con otra.
148     */
149    protected final Camino listaLocalidadesACamino(final List<Term> listaLocalidades) {
150        final Camino camino = new Camino(grafoCarreteras);
151        int i = 0;
152        Localidad localidadAnterior = null;
153        Localidad localidadActual = null;
154        final Iterator<Term> iter = listaLocalidades.iterator();
155
156        // Aunque quizás esta comprobación parezca defensiva de más, no lo es tanto: si un agente nos envía una
157        // variable no instanciada, puede interpretarse como una lista, pero que no tiene iterador (fantástica
158        // seguridad de tipos la de Jason)
159        if (iter != null) {
160            while (iter.hasNext()) {
161                final Term t = iter.next();
162
163                // Comprobar que la localidad sea del tipo esperado
164                if (t instanceof Atom) {
165                    localidadActual = gestorPercepciones.localidadAtomo((Atom) t);
166
167                    // Comprobar que realmente se corresponde con una localidad existente
168                    if (localidadActual != null) {
169                        // Añadir carreteras del camino a partir de la segunda localidad, enlazándola
170                        // con la inmediatamente anterior
171                        if (i > 0) {
172                            camino.addCarretera(localidadAnterior, localidadActual);
173                        }
174                    } else {
175                        throw new IllegalArgumentException("La localidad " + ((Atom) t).getFunctor() + " indicada por una hormiga no existe en el mapa actual.");
176                    }
177                } else {
178                    throw new IllegalArgumentException("Un elemento de la lista de localidades del camino encontrado por una hormiga no es un átomo.");
179                }
180
181                // Actualizar localidad inmediatamente anterior y contador de localidades recorridas
182                localidadAnterior = localidadActual;
183                ++i;
184            }
185        }
186
187        // Tiene poco sentido considerar caminos que no consistan en al menos una carretera
188        if (i < 2) {
189            throw new IllegalArgumentException("El camino seguido por una hormiga no tiene al menos dos localidades.");
190        }
191
192        // Hemos terminado de añadir carreteras al camino. Hacerlo de solo lectura
193        camino.terminarConstruccion();
194
195        return camino;
196    }

198    /**
199     * Comprueba si el camino especificado constituye una solución candidata del problema de búsqueda.
200     * @param camino El camino a comprobar si conduce a una solución. Se asume que es no nulo.
201     * @return Verdadero si y solo si el camino dado conduce a una solución del problema, falso en
202     * caso contrario.
203     */
204    protected final boolean conduceASolucion(final Camino camino) {
205        return localidadDestino.equals(camino.getUltimaLocalidad());
206    }

208    /**
209     * Señaliza la conclusión de una iteración. Como efectos colaterales, este método
210     * actualiza estructuras de datos internas cuyo contenido debe de variar por la finalización
211     * de una iteración.
212     */
213    protected final void finalizarIteracion() {
214        hormigasRecorridoArco.clear();
215    }
216
217    /**
218     * Avanza el algoritmo al siguiente ciclo. Como efectos colaterales, este método
219     * actualiza estructuras de datos internas cuyo contenido debe de variar por la
220     * finalización de un ciclo. Nótese que la conclusión de un ciclo implica la
221     * conclusión de una iteración: al realizar la última iteración de un ciclo,
222     * una llamada a este método finaliza automáticamente la iteración actual.
223     * Así pues, es innecesario (e incluso podría ser inseguro, desde un punto de vista
224     * de concurrencia) finalizar la última iteración de un ciclo explícitamente,
225     * mediante una llamada previa o posterior a {@link finalizarIteracion}.
226     */
227    protected final void siguienteCiclo() {
228        hormigasRecorridoArco.clear();
229        hormigasCaminoEncontrado.clear();
230        carreterasCaminoSolucion.clear();
231        ciclo.increment();
232    }

234    /**
235     * Consulta las estructuras de datos internas del algoritmo, que se mantienen actualizadas
236     * actualizadas con las apropiadas invocaciones de métodos señalizadores de eventos, para
237     * determinar si todas las hormigas han realizado el mismo camino solución en el ciclo actual.
238     * Cuando termine la invocación de este método, la estructura de datos interna que se encarga
239     * de registrar qué caminos encontraron las hormigas se vaciará, por lo que una segunda llamada
240     * a este método (u otros que dependan de esa información) puede dar un resultado diferente.
241     * En caso de que alguna subclase sobreescriba alguna implementación de algún método, debe de cumplir
242     * los términos del contrato aquí documentados para que este método siga funcionando correctamente.
243     * @return Verdadero si y solo si todas las hormigas han ido por el mismo camino solución en este ciclo.
244     */
245    protected final boolean todasHormigasHicieronMismoCamino() {
246        boolean toret = ciclo.intValue() > 1;
247        Camino caminoPrimero;
248        Camino caminoActual;
249
250        // Solo considerar caminos repetidos si no es la primera iteración. Puede pasar que,
251        // con pocas hormigas y localidades, tengamos la mala suerte de que vayan por el mismo
252        // camino la primera vez. Que lo hagan dos veces es ya bastante más raro
253        if (toret) {
254            // Tomar el primer elemento como el primer anterior. Si no hay un primer elemento,
255            // entonces no puede haber dos caminos iguales, por lo que debemos de devolver falso
256            caminoPrimero = caminosEncontrados.poll();
257            if (caminoPrimero != null) {
258                // Si tiene un segundo elemento, y ese segundo elemento es igual al primero,
259                // entonces seguir comparando hasta que lleguemos al final o encontremos un
260                // camino distinto. Si no tiene un segundo elemento, entonces ninguna hormiga
261                // hizo un camino o bien solo hay una hormiga en el sistema, con lo cual no
262                // se considera que varias hormigas hiciesen el mismo
263                caminoActual = caminosEncontrados.poll();
264                do {
265                    toret = caminoPrimero.equals(caminoActual);
266                    caminoActual = caminosEncontrados.poll();
267                } while (caminoActual != null && toret);
268            } else {
269                toret = false;
270            }
271        }
272
273        // Eliminar caminos restantes que quedasen
274        caminosEncontrados.clear();
275
276        return toret;
277    }

279    /**
280     * Comprueba si se debiera de avanzar al siguiente ciclo, dado el estado actual
281     * de las estructuras de datos internas del algoritmo usadas para deducir tal conclusión.
282     * @return Verdadero si y solo si se debiera de avanzar al siguiente ciclo, falso en
283     * caso contrario.
284     */
285    protected final boolean procedeAvanzarASigCiclo() {
286        // Cuando ninguna hormiga ha avanzado desde donde está, es porque ya todas están en el hormiguero,
287        // y han encontrado un camino (o todas están bloqueadas, lo que no debiera de ocurrir, y avanzar al
288        // siguiente ciclo quizás ayude a resolverlo)
289        return hormigasRecorridoArco.isEmpty();
290    }

302     * Realiza un postprocesamiento de todos los caminos solución encontrados por las hormigas en el ciclo actual,

310     * Realiza un postprocesamiento de un arco (carretera) recorrido por una hormiga en una iteración,

317    /**
318     * {@inheritDoc} En el caso de esta implementación, se toma nota de las hormigas que recorrieron un
319     * arco en la iteración actual. La mayoría de subclases no encontrarán utilidad a sobreescribir
320     * este método.
321     */
322    @Override
323    public void recorrerArco(final String hormiga, final Carretera carretera) {
324        hormigasRecorridoArco.add(hormiga);
325
326        // Avisar a subclases del arco recorrido
327        procesarArco(carretera);
328    }

330    /**
331     * {@inheritDoc} Esta implementación simplemente toma nota del camino recorrido por la hormiga,
332     * actualizando estructuras de datos internas. La mayoría de subclases no encontrarán utilidad a
333     * sobreescribir este método.
334     */
335    @Override
336    public void notificarCaminoEncontrado(final String hormiga, final List<Term> listaLocalidades) {
337        // Validar e interpretar parámetros de entrada
338        final Camino camino = listaLocalidadesACamino(listaLocalidades);
339
340        // Añadir a conjunto de hormigas que encontraron un camino ahora
341        final boolean caminoEncontradoAhora = hormigasCaminoEncontrado.add(hormiga);
342
343        // ¿Encontró un camino en el ciclo actual, y solo en el ciclo actual?
344        if (caminoEncontradoAhora) {
345            // Registrar el camino que encontró
346            caminosEncontrados.add(camino);
347
348            // Si conduce a una solución, registrarlo como tal
349            if (conduceASolucion(camino)) {
350                // Asociar el camino con las carreteras por las que pasa
351
352                // TODO: paralelizar con Spliterators
353                for (final Carretera carretera : camino) {
354                    carreterasCaminoSolucion.compute(carretera, (final Carretera ignorada, final List<Camino> caminosAnteriores) -> {
355                        List<Camino> toret = caminosAnteriores;
356
357                        // Crear una lista si todavía no tenemos una para esta carretera. No hace falta que sea
358                        // segura entre varios hilos de ejecución porque compute() y get() ya establecen relaciones
359                        // pasa-antes respecto a su modificación
360                        if (toret == null) {
361                            toret = new LinkedList<>();
362                        }
363                        toret.add(camino);
364
365                        return toret;
366                    });
367                }
368
369                // Si esta es una mejor solución que la que actualmente tenemos, registrarla como tal
370                synchronized (candadoMejorSolucionEncontrada) {
371                    if (mejorSolucionEncontrada == null || camino.compareTo(mejorSolucionEncontrada) < 0) {
372                        mejorSolucionEncontrada = camino;
373                    }
374                }
375            }
376        }
377    }

388    /**
389     * {@inheritDoc} Esta implementación simplemente actualiza el estado interno del algoritmo, sin cambiar
390     * los niveles de feromona de ninguna carretera. La mayoría de subclases no encontrarán utilidad a
391     * sobreescribir este método. Se asume que se llama a este método tras la notificación por parte de las
392     * hormigas de todos los eventos pertinentes, pues no tendría sentido razonar sobre información no
393     * completa de un episodio del SMA.
394     */
395    @Override
396    public boolean avanzarIteracion() {
397        boolean toret = procedeAvanzarASigCiclo();
398
399        // Si debemos de avanzar al siguiente ciclo, realizar las tareas pertinentes
400        if (toret) {
401            // Condiciones de terminación: hemos llegado al número de ciclos máximo, o todas
402            // las hormigas siguieron el mismo camino
403            if (getCiclo() >= getCiclosMaximos() - 1) {
404                terminar(RazonTerminacion.CICLOS_CONSUMIDOS);
405            } else if (todasHormigasHicieronMismoCamino()) {
406                terminar(RazonTerminacion.ESTANCAMIENTO);
407            }
408
409            // Realizar procesamiento de los caminos solución encontrados, por ejemplo para
410            // depositar feromonas (depende de la implementación)
411            procesarCaminosSolucionCiclo(carreterasCaminoSolucion);
412
413            // Finalmente, avanzar al siguiente ciclo
414            siguienteCiclo();
415        } else {
416            // Ir a la siguiente iteración
417            finalizarIteracion();
418        }
419
420        return toret;
421    }

443    @Override
444    public final Camino getMejorCamino() {
445        synchronized (candadoMejorSolucionEncontrada) {
446            if (mejorSolucionEncontrada != null) {
447                return mejorSolucionEncontrada;
448            } else {
449                throw new NoSuchElementException();
