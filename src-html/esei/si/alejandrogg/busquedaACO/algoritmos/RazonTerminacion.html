005/**
006 * Un enumerado que representa las posibles causas de terminación de un algoritmo de búsqueda
007 * basado en ACO.
008 * @author Alejandro González García
009 */
010public enum RazonTerminacion {
011    /**
012     * El algoritmo ha terminado, pero no se sabe por qué, o ninguna otra razón es apropiada.
013     * @deprecated Ningún código debería de usar esta causa normalmente. Se provee a modo de
014     * "placeholder" para pruebas, depuración u otros casos similares.
015     */
016    @Deprecated
017    DESCONOCIDA {
018        @Override
019        public String toString() {
020            return "causa de terminación desconocida";
021        }
022    },
023    /**
024     * El algoritmo ha terminado porque todas las hormigas han recorrido el mismo camino
025     * en el ciclo actual.
026     */
027    ESTANCAMIENTO {
028        @Override
029        public String toString() {
030            return "comportamiento estancado";
031        }
032    },
033    /**
034     * El algoritmo ha terminado porque ha consumido el número de ciclos máximos que podía hacer.
035     */
036    CICLOS_CONSUMIDOS {
037        @Override
038        public String toString() {
039            return "ciclos consumidos";
040        }
041    }
042}
