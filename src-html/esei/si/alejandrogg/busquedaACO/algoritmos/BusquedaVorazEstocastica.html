003package esei.si.alejandrogg.busquedaACO.algoritmos;
004
005import java.util.List;
006import java.util.Map;
007
008import esei.si.alejandrogg.busquedaACO.Camino;
009import esei.si.alejandrogg.busquedaACO.Carretera;
010import esei.si.alejandrogg.busquedaACO.Localidad;
011import esei.si.alejandrogg.busquedaACO.GestorPercepciones;
012import esei.si.alejandrogg.busquedaACO.GrafoCarreteras;
013
014/**
015 * Implementa un algoritmo para un "ant-system" que no deposita feromonas ni
016 * las evapora, por lo que esencialmente proporciona un comportamiento similar
017 * al de una búsqueda voraz estocástica que no tiene en cuenta experiencias anteriores.
018 * Cualquier otro algoritmo que sí deposite feromona debería de converger más rápidamente
019 * a las soluciones y centrar los esfuerzos de exploración en los caminos más prometedores,
020 * por lo que este algoritmo sencillo proporciona una línea base y pesimista respecto a
021 * la que comparar resultados.
022 * @author Alejandro González García
023 */
024final class BusquedaVorazEstocastica extends BaseAlgoritmo {
025    /**
026     * El nombre identificativo de este algoritmo.
027     */
028    static final String NOMBRE = "Búsqueda voraz estocástica";
029
030    /**
031     * Crea un nuevo algoritmo de búsqueda voraz estocástica a partir de los parámetros
032     * especificados.
033     * @param grafoCarreteras El grafo de carreteras sobre el que operará este algoritmo.
034     * @param gestorPercepciones El gestor de percepciones. Aunque este algoritmo en
035     * particular no modificará su entorno mediante este gestor, el parámetro se recibe y
036     * valida para proporiconar una interfaz común a otros algoritmos.
037     * @param ciclosMaximos Los ciclos máximos a ejecutar por el algoritmo.
038     * @param coeficienteRetencion Un parámetro ignorado por este algoritmo.
039     * @param q Un parámetro ignorado por este algoritmo.
040     * @param localidadDestino La localidad de destino, de la que el algoritmo intentará
041     * encontrar el camino más corto hacia ella.
042     * @throws IllegalArgumentException Si grafoCarreteras es nulo, gestorPercepciones es nulo,
043     * ciclosMaximos es menor que 1 o localidadDestino es nula.
044     */
045    BusquedaVorazEstocastica(final GrafoCarreteras grafoCarreteras, final GestorPercepciones gestorPercepciones, final int ciclosMaximos, final double coeficienteRetencion, final double q, final Localidad localidadDestino) {
046        super(grafoCarreteras, gestorPercepciones, ciclosMaximos, localidadDestino, NOMBRE);
047    }
048
049    /**
050     * No ejecuta ninguna instrucción.
051     * @param carretera Un parámetro ignorado en esta implementación.
052     */
053    @Override
054    public void procesarArco(final Carretera carretera) {}
055
056    /**
057     * No ejecuta ninguna instrucción.
058     * @param caminosCarretera Un parámetro ignorado en esta implementación.
059     */
060    @Override
061    public void procesarCaminosSolucionCiclo(final Map<Carretera, List<Camino>> caminosCarretera) {}
062}
