003package esei.si.alejandrogg.busquedaACO.algoritmos;
004
005import java.util.List;
006import java.util.Map;
007
008import esei.si.alejandrogg.busquedaACO.Camino;
009import esei.si.alejandrogg.busquedaACO.Carretera;
010import esei.si.alejandrogg.busquedaACO.Localidad;
011import esei.si.alejandrogg.busquedaACO.GestorPercepciones;
012import esei.si.alejandrogg.busquedaACO.GrafoCarreteras;
013
014/**
015 * Implementa un algoritmo para un "ant-system" que deposita feromonas
016 * solo cuando todas las hormigas involucradas han completado un camino
017 * (es decir, se avanza al siguiente ciclo). Este algoritmo está descrito en
018 * el artículo de Dorigo et al. (1996), "The Ant System: Optimization by a colony
019 * of cooperating agents". La implementación aquí hecha está paralelizada.
020 * @author Alejandro González García
021 */
022final class AntCycle extends BaseAlgoritmo {
023    /**
024     * El nombre identificativo de este algoritmo.
025     */
026    static final String NOMBRE = "Dorigo et al. (1996), ant-cycle";
027
028    /**
029     * El coeficiente de retención de feromona.
030     */
031    private final double coeficienteRetencion;
032    /**
033     * Una constante positiva lo más pequeña posible, que indica la magnitud de feromona a sumar
034     * en cada carretera por cada hormiga. Su valor debe ajustarse teniendo en cuenta la longitud
035     * máxima estimada de un camino solución, el coeficiente de retención, el valor subjetivo de
036     * importancia de feromona y el creciente error de precisión en el que incurre la representación
037     * de coma flotante IEEE 754 de 64 bits según las magnitudes de feromona se alejan del cero.
038     */
039    private final double q;
040
041    /**
042     * Crea un nuevo algoritmo "ant-cycle" a partir de los parámetros especificados.
043     * @param grafoCarreteras El grafo de carreteras sobre el que operará este algoritmo.
044     * @param gestorPercepciones El gestor de percepciones, que proporciona funcionalidades
045     * para actualizar el estado del entorno en base a los resultados del algoritmo.
046     * @param ciclosMaximos Los ciclos máximos a ejecutar por el algoritmo.
047     * @param coeficienteRetencion El coeficiente de retención de feromona a emplear.
048     * @param q Una constante positiva lo más pequeña posible, que influye en las magnitudes
049     * de feromona a sumar.
050     * @param localidadDestino La localidad de destino, de la que el algoritmo intentará
051     * encontrar el camino más corto hacia ella.
052     * @throws IllegalArgumentException Si grafoCarreteras es nulo, gestorPercepciones es nulo,
053     * ciclosMaximos es menor que 1, el coeficiente de retención no está en el intervalo [0, 1],
054     * q es menor o igual a 0 o localidadDestino es nula.
055     */
056    AntCycle(final GrafoCarreteras grafoCarreteras, final GestorPercepciones gestorPercepciones, final int ciclosMaximos, final double coeficienteRetencion, final double q, final Localidad localidadDestino) {
057        super(grafoCarreteras, gestorPercepciones, ciclosMaximos, localidadDestino, NOMBRE);
058
059        if (coeficienteRetencion < 0 || coeficienteRetencion > 1) {
060            throw new IllegalArgumentException("El coeficiente de retención de feromona a emplear no está en el intervalo [0, 1].");
061        }
062        if (q <= 0) {
063            throw new IllegalArgumentException("Q no puede tener un valor menor o igual a 0.");
064        }
065
066        this.coeficienteRetencion = coeficienteRetencion;
067        this.q = q;
068    }
069
070    /**
071     * No ejecuta ninguna instrucción.
072     * @param carretera Un parámetro ignorado en esta implementación.
073     */
074    @Override
075    public void procesarArco(final Carretera carretera) {}
076
077    /**
078     * Evapora las feromonas presentes en todas las carreteras, de manera que el nuevo nivel de feromona de cada carretera
079     * sea t = ρ · t, donde ρ ∈ [0, 1] es el coeficiente de retención del algoritmo, y a continuación simula el depósito de feromonas
080     * de las hormigas en las carreteras por las que pasaron, de manera que el nuevo valor de t de tales carreteras sea
081     * t = t + Δt, siendo Δt = Q / L, donde Q es una constante real positiva pequeña y L la distancia del camino solución recorrido
082     * por cada hormiga.
083     * @param caminosCarretera El mapa que relaciona cada carretera con los caminos solución que pasan por ella.
084     */
085    @Override
086    public void procesarCaminosSolucionCiclo(final Map<Carretera, List<Camino>> caminosCarretera) {
087        // TODO: paralelizar con Spliterators
088        for (final Carretera carretera : grafoCarreteras.carreteras()) {
089            // Evaporar feromona, teniendo en cuenta el coeficiente de retención
090            carretera.actualizarNivelFeromona(gestorPercepciones, (final Double t) -> {
091                return coeficienteRetencion * t;
092            });
093
094            // Para los caminos que pasan por aquí, depositar feromonas de acuerdo a sus características
095            final List<Camino> caminos = caminosCarretera.get(carretera);
096            if (caminos != null) {
097                for (final Camino camino : caminos) {
098                    carretera.actualizarNivelFeromona(gestorPercepciones, (final Double t) -> {
099                        final double deltaT = q / camino.distanciaTotal();
100                        return t + deltaT;
101                    });
102                }
103            }
104        }
105    }
106}
