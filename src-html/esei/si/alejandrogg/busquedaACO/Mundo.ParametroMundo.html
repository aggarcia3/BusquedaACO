196    /**
197     * El coordinador de hormigas en uso por el mundo, que provee servicios de coordinación
198     * entre hormigas, para garantizar la consistencia de las invariantes del sistema.
199     */
200    private final CoordinadorHormigas coordinadorHormigas = new CoordinadorHormigas(this, (final String nombre) -> {
201        // Incrementar el número de hormigas presentes en el sistema
202        nHormigas.increment();
203
204        // Añadir la hormiga a la localidad de inicio
205        getLocalizadorHormigas().mover(nombre, locInicio);
206
207        // Si no estamos esperando a que lleguen las hormigas, actualizar la vista
208        // La vista puede ser nula si el SMA se está deteniendo por un error, pero llega una hormiga mientras tanto
209        if (vista != null && !getCoordinadorHormigas().esperandoLlegadaHormigas()) {
210            vista.actualizar();
211        }
212
213        getLogger().log(Level.INFO, "Detectada nueva hormiga en el sistema: " + nombre);
214    }, (final String nombre) -> {
215        // Decrementar el número de hormigas presentes en el sistema
216        nHormigas.decrement();
217
218        // Eliminar a la hormiga de donde quiera que esté
219        getLocalizadorHormigas().desaparecer(nombre);
220
221        // Si hemos cargado un algoritmo desde la configuración, notificarle la ida
222        if (algoritmo != null) {
223            algoritmo.notificarMuerteHormiga(nombre);
224        }
225
226        // Actualizar la vista, para mostrar el nuevo número de hormigas
227        if (vista != null) {
228            vista.actualizar();
229        }
230
231        getLogger().log(Level.INFO, "Una hormiga deja de formar parte del sistema: " + nombre);
232    });

439        /**
440         * Interpreta un parámetro de configuración representado por un número real positivo y finito,
441         * a partir de una cadena de texto, mostrando un mensaje de error al usuario si el número no es tal.
442         * @param m El mundo vinculado con este parámetro de configuración.
443         * @param arg El parámetro de configuración expresado como una cadena de texto.
444         * @param mensajeError El mensaje de error a mostrar al usuario si el número no es real, positivo
445         * y finito.
446         * @param permitirMayorQueUno Si es verdadero, se considerarán inválidos números reales estrictamente
447         * mayores que uno. En caso contrario, se tomarán por válidos.
448         * @param permitirCero Si es verdadero, se considerará válido el cero. En caso contrario, el cero no
449         * será un valor válido.
450         * @return El número real interpretado.
451         * @throws IllegalArgumentException Si el parámetro no es un número real, positivo y finito.
452         */
453        final protected double interpretarReal(final Mundo m, final String arg, final String mensajeError, final boolean permitirMayorQueUno, final boolean permitirCero) {
454            try {
455                final double toret = Double.parseDouble(arg);
456
457                // También rechazar números negativos, iguales a cero si procede, mayores que uno si procede, infinitos o NaN
458                if (toret < 0 || (!permitirCero && toret == 0) || (!permitirMayorQueUno && toret > 1) || Double.isInfinite(toret) || Double.isNaN(toret)) {
459                    throw new NumberFormatException();
460                }
461
462                return toret;
463            } catch (NumberFormatException exc) {
464                m.getLogger().log(Level.SEVERE, mensajeError);
465                throw new IllegalArgumentException();
466            }
467        }

537    /**
538     * Obtiene la única instancia de esta clase (patrón singleton). Se garantiza que el
539     * resultado de la ejecución de este método nunca es nulo, y devuelve un objeto
540     * totalmente inicializado, sin atributos visiblemente nulos.
541     * @return La devandicha instancia.
542     */
543    public static Mundo get() {
544        // Esperar a que la instancia del mundo termine de inicializarse
545        candadoInicializacion.lock();
546        try {
547            while (instancia == null) {
548                try {
549                    final boolean tiempoNoExpirado = mundoInicializado.await(15, TimeUnit.SECONDS);
550                    if (!tiempoNoExpirado) {
551                        // Avisar al usuario de que podríamos estar bloqueados
552                        Logger.getLogger("EnvironmentWatchdog").log(Level.WARNING, "El entorno no se ha inicializado en 15 segundos. Si la espera tarda demasiado, por favor investigue la causa y reinicie el sistema.");
553                    }
554                } catch (InterruptedException exc) {
555                    Logger.getLogger("EnvironmentWatchdog").log(Level.INFO, "Esperando a que se inicie el entorno...");
556                }
557            }
558        } finally {
559            candadoInicializacion.unlock();
560        }
561
562        return instancia;
563    }

565    /**
566     * Punto de entrada del entorno, ejecutado antes del comienzo del SMA.
567     * @param args Los argumentos especificados en el fichero .mas2j. Véase la
568     * documentación de {@link interpretarArgumentos} para su significado.
569     */
570    @Override
571    public void init(final String[] args) {
572        boolean detenerSMA = false;
573
574        // Obtener referencia al objeto con servicios útiles para controlar el SMA
575        rs = getEnvironmentInfraTier() != null ? getEnvironmentInfraTier().getRuntimeServices() : null;
576
577        if (!interpretarArgumentos(args)) {
578            getLogger().log(Level.SEVERE, "Ha ocurrido un error interpretando algún parámetro proporcionado al mundo.");
579            detenerSMA = true;
580        }
581
582        // Avisar al usuario si no tenemos control sobre el SMA por algún motivo
583        if (rs == null) {
584            getLogger().log(Level.SEVERE, "El entorno no pudo obtener una referencia a los servicios de tiempo de ejecución de la infraestructura del SMA, lo que significa que no tiene control sobre él y no funcionará correctamente. Por favor, reinicia la aplicación.");
585            detenerSMA = true;
586        }
587
588        if (!detenerSMA) {
589            // Esperamos a que se creen las hormigas iniciales. Si llegasen hormigas al sistema
590            // antes o durante la inicialización del entorno (lo que puede ocurrir en la arquitectura
591            // de JADE), su notificación de llegada se retendría hasta que se finalice la ejecución de
592            // este método
593            coordinadorHormigas.esperarLlegadaHormigas(() -> {
594                // Actualizar la vista para mostrar el estado inicial. Puede
595                // ser nula en caso de que el hilo que ejecuta init() sufra
596                // exagerada inanición, y la espera por la llegada de las hormigas
597                // se complete antes que la creación de la vista
598                if (vista != null) {
599                    vista.actualizar();
600                }
601
602                // Empezar el avance del tiempo discreto de la simulación
603                tiempo = new Tiempo(Mundo.this, () -> {
604                    boolean pararAvanceTiempo = false;
605
606                    // Notificar a los agentes del paso del tiempo, y esperar a que hagan lo que tengan que hacer
607                    coordinadorHormigas.esperarRespuestaHormigas(() -> gp.percibirPasoTiempo(), segundosEsperaRespuesta);
608
609                    // Ahora, cuando todas las hormigas respondieron, es la hora de ejecutar
610                    // las acciones del entorno
611                    final long inicioEjecucionAcciones = System.currentTimeMillis();
612                    getLogger().log(Level.INFO, "Ejecutando acciones sobre el entorno enviadas por las hormigas...");
613
614                    AccionProgramada accion = accionesProgramadas.poll();
615                    while (accion != null) {
616                        // Registrar una nueva acción a ejecutar
617                        ejecucionAcciones.register();
618
619                        // Realizar la acción en otro hilo de ejecución
620                        servicioAccionesEntorno.execute(new TareaAccionEntorno(accion));
621
622                        // Consumir la siguiente acción en la cola
623                        accion = accionesProgramadas.poll();
624                    }
625
626                    // Esperar hasta que nos interrumpan o las acciones terminen de ejecutarse
627                    boolean esperaCompletada = false;
628                    while (!esperaCompletada) {
629                        try {
630                            ejecucionAcciones.awaitAdvanceInterruptibly(ejecucionAcciones.arrive(), 30, TimeUnit.SECONDS);
631                            esperaCompletada = true;
632                        } catch (TimeoutException exc) {
633                            getLogger().log(Level.WARNING, "Se está tardando más de 30 segundos en ejecutar las acciones del entorno programadas. Si la espera tarda demasiado, por favor reinicie el sistema o interrumpa este hilo.");
634                        } catch (InterruptedException exc) {
635                            getLogger().log(Level.WARNING, "Se ha interrumpido el hilo que esperaba a que finalizase la ejecución de las acciones en el entorno. El hilo hará caso a la interrupción y abandonará la espera, pero ello puede volver inestable al sistema.");
636                            pararAvanceTiempo = true;
637                        }
638                    }
639
640                    getLogger().log(Level.INFO, "Acciones sobre el entorno ejecutadas en " + ((System.currentTimeMillis() - inicioEjecucionAcciones) / 1000.0) + " s.");
641
642                    // Avanzamos a la siguiente iteración del algoritmo
643                    final long inicioAvanceIteracion = System.currentTimeMillis();
644                    getLogger().log(Level.INFO, "Avanzando a la siguiente iteración...");
645                    boolean avanceNuevoCiclo = false;
646                    try {
647                        avanceNuevoCiclo = algoritmo.avanzarIteracion();
648                    } catch (Exception exc) {
649                        getLogger().log(Level.WARNING, "Ha ocurrido una excepción no controlada mientras el algoritmo avanzaba a la siguiente iteración.", exc);
650                    }
651                    getLogger().log(Level.INFO, "Se ha avanzado a la siguiente iteración en " + ((System.currentTimeMillis() - inicioAvanceIteracion) / 1000.0) + " s.");
652
653                    // En caso de que se empiece un nuevo ciclo, mover todas las hormigas
654                    // a la localidad de inicio
655                    if (avanceNuevoCiclo) {
656                        final long inicioAvanceCiclo = System.currentTimeMillis();
657                        getLogger().log(Level.INFO, "El algoritmo indica el comienzo de un nuevo ciclo. Avisando a hormigas del comienzo del nuevo ciclo...");
658
659                        // Ir ejecutando el movimiento de todas las hormigas a la localidad de inicio
660                        // en otro hilo. De esta forma podríamos hacer algo útil mientras esperamos
661                        // por las respuestas a las percepciones
662                        final Future<?> terminacionMovimiento = servicioDesplazamientoHormigas.submit(() -> {
663                            localizadorHormigas.moverTodasALocalidad(locInicio);
664                        });
665
666                        // Esperar a que informen de la recepción de la percepción
667                        coordinadorHormigas.esperarRespuestaHormigas(() -> {
668                            gp.siguienteCiclo();
669                        }, segundosEsperaRespuesta);
670
671                        // Esperar a que se terminen de mover las hormigas a la localidad de inicio:
672                        // nada garantiza que ello se termine antes que la respuesta de los agentes
673                        try {
674                            terminacionMovimiento.get();
675                        } catch (ExecutionException exc) {
676                            getLogger().log(Level.WARNING, "Ha ocurrido una excepción no controlada mientras se movían hormigas a la localidad de inicio. El sistema no funcionará correctamente a partir de ahora, y se recomienda su detención. Detalles: ", exc);
677                        } catch (InterruptedException exc) {
678                            getLogger().log(Level.WARNING, "Se ha interrumpido la espera a que todas las hormigas se colocasen de nuevo en la localidad de inicio.");
679                            pararAvanceTiempo = true;
680                        }
681
682                        getLogger().log(Level.INFO, "Avance de ciclo completado en " + ((System.currentTimeMillis() - inicioAvanceCiclo) / 1000.0) + " s.");
683                    }
684
685                    // Si el algoritmo ha terminado, dejar de avanzar el tiempo, pues no hay más que hacer
686                    final RazonTerminacion razonTerminacion = algoritmo.razonTerminacion();
687                    if (razonTerminacion != null) {
688                        getLogger().log(Level.INFO, "El algoritmo ha terminado de ejecutarse: " + razonTerminacion + ".");
689                        pararAvanceTiempo = true;
690                    }
691
692                    // Finalmente, actualizar la vista mostrada al usuario
693                    if (vista != null) {
694                        vista.actualizar();
695                    }
696
697                    return pararAvanceTiempo;
698                }, milisegundosPeriodoTiempo).iniciar();
699            }, segundosEsperaInicio);
700
701            // Avisar a hilos interesados de que nos hemos inicializado, poniendo a su disposición
702            // la referencia a esta instancia
703            candadoInicializacion.lock();
704            try {
705                mundoInicializado.signalAll();
706                instancia = this;
707            } finally {
708                candadoInicializacion.unlock();
709            }
710
711            // Inicializar la vista asociada al mundo. Lo hacemos algo tarde, aún tras
712            // avisar a otros hilos de que hemos completado la inicialización del Mundo,
713            // para que la vista pueda obtener una referencia al Mundo sin bloquearse. Además,
714            // hacerlo algo tarde no es un problema en la implementación actual, porque el Mundo
715            // nunca proporciona una referencia a la vista a otros objetos, gracias a la encapsulación :)
716            try {
717                vista = new VistaGrafo(grafoCarreteras, grafoCarreteras.getNombre() + ".png");
718            } catch (NullPointerException | IllegalArgumentException | IllegalStateException exc) {
719                detenerSMA = true;
720            }
721        }
722
723        // Detener el SMA si procede
724        if (detenerSMA && rs != null) {
725            getLogger().log(Level.WARNING, "Deteniendo SMA.");
726
727            // Esperar un poco antes de detener el sistema, para darle tiempo al usuario de ver qué pasó
728            Executors.newSingleThreadScheduledExecutor().schedule(() -> {
729                try {
730                    rs.stopMAS();
731                } catch (Exception ignorada) {}
732            }, 10, TimeUnit.SECONDS);
733
734            // Lanzarle una excepción a quien nos ha llamado. En la arquitectura centralizada, esto evita
735            // la creación de agentes. En Jade no hace nada, pero da a entender nuestra intención
736            throw new IllegalStateException("El SMA debe de detenerse; el entorno no se pudo inicializar.");
737        }
738    }

740    /**
741     * Interpreta los argumentos especificados para el entorno en el fichero .mas2j,
742     * inicializando atributos de este objeto.
743     * @param args Los argumentos leídos del fichero.
744     * @return Verdadero si y solo si todos los argumentos se han leído correctamente, falso en
745     * otro caso.
746     */
747    private boolean interpretarArgumentos(final String[] args) {
748        final ParametroMundo[] params = ParametroMundo.values();
749        boolean toret = args != null && args.length == params.length;
750
751        if (toret) {
752            // Interpretar todos los argumentos recibidos mientras no ocurra un error
753            for (int i = 0; i < args.length && toret; ++i) {
754                String arg = args[i];
755
756                // Considerar argumentos nulos como erróneos
757                if (arg == null) {
758                    getLogger().log(Level.SEVERE, "El argumento " + (i + 1) + " pasado al mundo es nulo.");
759                    toret = false;
760                } else {
761                    toret = params[i].interpretar(this, arg);
762                }
763            }
764        } else {
765            getLogger().log(Level.SEVERE, "No se ha recibido el número de parámetros esperado para inicializar el mundo.");
766        }
767
768        return toret;
769    }

771    /**
772     * Comprueba si un conjunto de localidades contiene una determinada localidad, identificada
773     * por su nombre, y de ser así devuelve tal localidad.
774     * @param localidades Las localidades donde buscar una localidad en particular.
775     * @param localidad La localidad a intentar encontrar.
776     * @return La localidad especificada si se pudo encontrar, nulo en otro caso.
777     */
778    private static Localidad contieneLocalidad(final Set<Localidad> localidades, final String localidad) {
779        Localidad toret = null;
780        final Iterator<Localidad> iter = localidades.iterator();
781
782        while (toret == null && iter.hasNext()) {
783            final Localidad actual = iter.next();
784            if (actual.getNombre().equals(localidad)) {
785                toret = actual;
786            }
787        }
788
789        return toret;
790    }

884    @Override
885    public void scheduleAction(final String agName, final Structure action, final Object infraData) {
886        boolean ejecucionExitosa = true;
887
888        try {
889            // Solo programar la ejecución si es una acción válida
890            if (operadorAcciones.esAccionValida(agName, action)) {
891                getLogger().log(Level.FINE, "La hormiga " + agName + " envió la respuesta " + action.getFunctor() + ".");
892
893                // Añadir a la cola de acciones programadas
894                accionesProgramadas.add(new AccionProgramada(agName, action));
895
896                // Finalmente, podemos decir que la hormiga ha ejecutado una respuesta
897                coordinadorHormigas.respuestaHormiga(agName);
898            } else {
899                throw new UnsupportedOperationException("La acción de entorno \"" + (action != null ? action.getFunctor() : "(nula)") + "\" no existe, o no puede ser ejecutada por " + agName + " en este momento.");
900            }
901        } catch (Exception exc) {
902            // Registrar que se ha producido un error
903            getLogger().log(Level.WARNING, "Ha ocurrido una excepción no controlada durante la programación de una acción de entorno: ", exc);
904            ejecucionExitosa = false;
905        } finally {
906            // Avisar a la infraestructura de que la ejecución de la acción ha terminado
907            getEnvironmentInfraTier().actionExecuted(agName, action, ejecucionExitosa, infraData);
908        }
909    }

911    /**
912     * Representa una acción programada para su ejecución en el entorno, a cargo de una hormiga.
913     * @author Alejandro González García
914     */
915    private static final class AccionProgramada {
916        /**
917         * La hormiga responsable de la acción.
918         */
919        private final String hormiga;
920        /**
921         * La acción que la hormiga desea ejecutar.
922         */
923        private final Structure accion;
924
925        /**
926         * Crea una nueva acción programada a cargo de una hormiga.
927         * @param hormiga La hormiga responsable.
928         * @param accion La acción que desea ejecutar.
929         */
930        AccionProgramada(final String hormiga, final Structure accion) {
931            this.hormiga = hormiga;
932            this.accion = accion;
933        }
934
935        /**
936         * Obtiene la hormiga responsable de esta acción.
937         * @return La devandicha hormiga.
938         */
939        String getHormiga() {
940            return hormiga;
941        }
942
943        /**
944         * Obtiene la acción que la hormiga pretende ejecutar.
945         * @return La devandicha acción.
946         */
947        Structure getAccion() {
948            return accion;
949        }

952    /**
953     * Implementa la lógica necesaria para ejecutar una {@link AccionProgramada}
954     * determinada en otro hilo de ejecución.
955     * @author Alejandro González García
956     */
957    private final class TareaAccionEntorno implements Runnable {
958        /**
959         * La acción programada a ejecutar.
960         */
961        private final AccionProgramada accion;
962
963        /**
964         * Crea una nueva tarea para ejecutar una acción programada determinada,
965         * opcionalmente en otro hilo.
966         * @param accion La acción a ejecutar. El constructor no comprueba que no
967         * sea nula.
968         */
969        TareaAccionEntorno(final AccionProgramada accion) {
970            this.accion = accion;
971        }
972
973        @Override
974        public void run() {
975            // Realizar la acción propiamente dicha
976            final String hormiga = accion.getHormiga();
977            final Structure estructuraAccion = accion.getAccion();
978            try {
979                operadorAcciones.ejecutarAccion(hormiga, estructuraAccion);
980                getLogger().log(Level.FINE, "La hormiga " + hormiga + " ha ejecutado la acción " + estructuraAccion + " sobre el entorno.");
981            } catch (Exception exc) {
982                getLogger().log(Level.WARNING, "Ha ocurrido una excepción no controlada durante la ejecución de una acción en el entorno para la hormiga " + hormiga + ". Se continúa la ejecución del SMA igualmente. Detalles: ", exc);
983            }
984
985            // Hemos terminado de hacerla
986            ejecucionAcciones.arriveAndDeregister();
987        }
988    }
