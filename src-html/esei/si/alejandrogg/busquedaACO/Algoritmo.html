003package esei.si.alejandrogg.busquedaACO;
004
005import java.util.List;
006import java.util.NoSuchElementException;
007
008import jason.asSyntax.Term;
009
010import esei.si.alejandrogg.busquedaACO.algoritmos.RazonTerminacion;
011
012/**
013 * Define el contrato que debe de cumplir cualquier algoritmo de un SMA de búsqueda
014 * mediante ACO. En general, la implementación de un algoritmo es la encargada de dictaminar
015 * cuándo y cómo se actualizan las feromonas de las carreteras, en respuesta a las acciones
016 * realizadas por las hormigas. También mantiene el control sobre la terminación y las
017 * iteraciones del algoritmo.
018 *
019 * Las implementaciones de esta interfaz deben de garantizar la
020 * corrección de su implementación aún bajo diferentes hilos de ejecución.
021 *
022 * @author Alejandro González García
023 */
024public interface Algoritmo {
025    /**
026     * Informa al algoritmo de que una hormiga acaba de recorrer un arco. La respuesta a esta
027     * acción depende del algoritmo, pero normalmente implica actualizar la cantidad de feromona
028     * depositada en la carretera correspondiente. En general, debe de asumirse que este método puede
029     * realizar cualquier acción en su implementación.
030     * @param hormiga La hormiga que ha recorrido un arco (carretera). Las implementaciones pueden
031     * asumir que este parámetro no es nulo.
032     * @param carretera La carretera que ha tomado la hormiga. Las implementaciones pueden asumir que
033     * este parámetro no es nulo.
034     */
035    public void recorrerArco(final String hormiga, final Carretera carretera);
036
037    /**
038     * Notifica al algoritmo de que una hormiga ha terminado de realizar un camino, bien porque
039     * ha llegado al destino, o bien porque no puede seguir avanzando. La respuesta a esta acción
040     * depende del algoritmo. En general, debe de asumirse que este método puede realizar cualquier
041     * acción en su implementación.
042     * @param hormiga La hormiga que acaba de terminar un camino. Las implementaciones pueden
043     * asumir que este parámetro no es nulo.
044     * @param camino El camino realizado por la hormiga, en forma de lista de ciudades (nodos) por
045     * las que pasó. Las implementaciones pueden asumir que este parámetro no es nulo, pero no que
046     * los términos contenidos en la lista sean válidos. Tampoco pueden asumir que el acceso
047     * aleatorio a elementos de la lista sea de complejidad constante.
048     * @throws IllegalArgumentException Si algún término contenido en la lista no es un átomo que
049     * represente una localidad.
050     */
051    public void notificarCaminoEncontrado(final String hormiga, final List<Term> camino);
052
053    /**
054     * Informa al algoritmo de que una hormiga ha dejado de formar parte del sistema o, más generalmente,
055     * que no debe de ser tenida en cuenta a efectos del algoritmo. Las implementaciones son libres
056     * de utilizar esta información para lo que deseen, o de incluso ignorarla.
057     * @param hormiga La hormiga que ha dejado de formar parte del sistema.
058     */
059    public void notificarMuerteHormiga(final String hormiga);
060
061    /**
062     * Avanza a la siguiente iteración del SMA. Este método debe de ser invocado en cuanto se hayan
063     * recibido las respuestas de todas las hormigas, y en consecuencia invocado otros métodos del
064     * objeto algoritmo según correspondiese. Como parte del avance de la iteración, la implementación
065     * puede usar información deducida a partir de llamadas a otros métodos para cambiar su estado interno,
066     * recalculando, por ejemplo, si ha terminado o no, si hay que empezar un nuevo ciclo, etc.
067     * @return Verdadero si y solo si, aparte de avanzarse a la siguiente iteración, también se avanzó al
068     * siguiente ciclo.
069     */
070    public boolean avanzarIteracion();
071
072    /**
073     * Comprueba si el algoritmo ha terminado de ejecutarse, obteniendo la razón por la que ello ocurrió.
074     * Puede deberse a cualquiera de las razones representadas por {@link RazonTerminacion}.
075     * @return La razón de terminación del algoritmo, si ha terminado, o nulo en caso de que todavía no haya terminado.
076     */
077    public RazonTerminacion razonTerminacion();
078
079    /**
080     * Obtiene el número de ciclo en el que se encuentra actualmente la ejecución del algoritmo. El primer ciclo es
081     * el 1.
082     * @return El devandicho número.
083     */
084    public int getCiclo();
085
086    /**
087     * Obtiene el número máximo de ciclos que el algoritmo ejecutará.
088     * @return El devandicho número.
089     */
090    public int getCiclosMaximos();
091
092    /**
093     * Obtiene el mejor camino hacia el destino encontrado hasta el momento por el algoritmo.
094     * @return El devandicho camino.
095     * @throws NoSuchElementException Si el algoritmo todavía no conoce al menos un camino.
096     */
097    public Camino getMejorCamino();
098
099    /**
100     * Obtiene el nombre identificativo de este algoritmo, que se mostrará al usuario, y se usará para
101     * identificarlo en los parámetros de configuración del SMA.
102     * @return El devandicho nombre.
103     */
104    public String getNombre();
105}
