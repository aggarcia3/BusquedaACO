003import java.util.List;
004
005import jason.asSemantics.Agent;
006import jason.asSemantics.Message;
007
008/**
009 * Modela un agente hormiga, que es muy similar a un agente convencional,
010 * pero no acepta comunicación por mensajes de otros agentes, y tampoco
011 * se deja matar por sus congéneres.
012 * @author Alejandro González García
013 */
014public final class Hormiga extends Agent {
015    @Override
016    public boolean socAcc(final Message m) {
017        return false;
018    }
019
020    @Override
021    public boolean killAcc(final String agente) {
022        return !agente.startsWith("hormiga");
023    }
024}
