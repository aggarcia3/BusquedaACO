001package esei.si.alejandrogg.busquedaACO.agentes;
002
003import jason.architecture.AgArch;
004
005import esei.si.alejandrogg.busquedaACO.Mundo;
006
007/**
008 * Modela la arquitectura de agente hormiga, que es muy similar a una
009 * convencional, pero avisa al entorno en caso de que el agente
010 * se inicie, muera o se detenga.
011 * @author Alejandro González García
012 */
013public final class ArquitecturaHormiga extends AgArch {
014    /**
015     * {@inheritDoc}
016     * @implNote. El método asume que el entorno es accesible desde la JVM local.
017     * Para funcionar en una arquitectura distribuida, requiere ser reimplementado con
018     * la <a href="https://docs.oracle.com/javase/8/docs/technotes/guides/rmi/">API de RMI de Java</a>.
019     */
020    @Override
021    public void init() throws Exception {
022        super.init();
023
024        // Avisar al coordinador de nuestra llegada
025        Mundo.get().getCoordinadorHormigas().llegadaHormiga(getAgName());
026    }
027
028    /**
029     * {@inheritDoc}
030     * @implNote. El método asume que el entorno es accesible desde la JVM local.
031     * Para funcionar en una arquitectura distribuida, requiere ser reimplementado con
032     * la <a href="https://docs.oracle.com/javase/8/docs/technotes/guides/rmi/">API de RMI de Java</a>.
033     */
034    @Override
035    public void stop() {
036        super.stop();
037
038        // Avisar al coordinador de nuestra muerte
039        Mundo.get().getCoordinadorHormigas().muerteHormiga(getAgName());
040    }
041}
