003package esei.si.alejandrogg.busquedaACO.accionesInternas;

005import java.util.List;
006import java.util.ArrayList;
007
008import jason.JasonException;
009import jason.asSemantics.DefaultInternalAction;
010import jason.asSemantics.InternalAction;
011import jason.asSemantics.TransitionSystem;
012import jason.asSemantics.Unifier;
013import jason.asSyntax.Term;
014import jason.asSyntax.NumberTerm;
015import jason.asSyntax.NumberTermImpl;
016import jason.asSyntax.ListTerm;
017
018import esei.si.alejandrogg.busquedaACO.Mundo;
019import esei.si.alejandrogg.busquedaACO.agentes.ArquitecturaHormiga;
020
021/**
022 * Obtiene una muestra aleatoria de una distribución discreta dada, definida por su función
023 * masa de probabilidad.
024 * @serial exclude
025 */
026public class muestraDistribucionDiscreta extends DefaultInternalAction {
027    /**
028     * La única instancia existente en la JVM de esta acción interna (patrón singleton).
029     */
030    private static InternalAction instancia = null;
031
032    /**
033     * Obtiene, creando si y solo si es necesario, la única instancia de esta clase.
034     * @return La única instancia de esta clase.
035     */
036    public static InternalAction create() {
037        if (instancia == null) {
038            instancia = new muestraDistribucionDiscreta();
039        }
040        return instancia;
041    }
042
043    @Override
044    public int getMinArgs() {
045        return 2;
046    }
047
048    @Override
049    public int getMaxArgs() {
050        return 2;
051    }
052
053    @Override
054    protected void checkArguments(final Term[] args) throws JasonException {
055        super.checkArguments(args);
056
057        if (!args[0].isList()) {
058            throw JasonException.createWrongArgument(this, "El primer argumento debe de ser una función masa de probabilidad.");
059        }
060    }
061
062    @Override
063    public Object execute(final TransitionSystem ts, final Unifier un, final Term[] args) throws Exception {
064        boolean toret = true;
065
066        checkArguments(args);
067
068        if (args[0] instanceof ListTerm) {
069            // Lista para albergar los valores de la función masa de probabilidad
070            final List<Double> f = new ArrayList<>(Mundo.get().getNumeroLocalidades()); // Como mucho, tendremos n valores para las n localidades
071
072            // Reinterpretar los términos de la lista como números, si es posible, y añadirlos
073            // a la lista de valores de la función masa de probabilidad
074            for (Term t : (ListTerm) args[0]) {
075                if (t.isNumeric()) {
076                    f.add(((NumberTerm) t).solve());
077                } else {
078                    throw new JasonException("Un elemento de la función masa de probabilidad no es un número.");
079                }
080            }
081
082            // Unificar el segundo parámetro con el resultado de la muestra
083            toret = un.unifies(new NumberTermImpl(new DistribProbabilidadDiscreta(f).next()), args[1]);
084        } else {
085            // Esto no debería de ocurrir bajo condiciones normales de ejecución
086            throw new AssertionError();
087        }
088
089        return toret;
090    }
091}
