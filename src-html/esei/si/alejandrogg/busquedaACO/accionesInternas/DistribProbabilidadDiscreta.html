010/**
011 * Modela una distribución de probabilidad discreta, de la que se puede obtener
012 * un número natural aleatorio K con probabilidad pk = P(K), donde P(K) es el valor
013 * correspondiente de la función masa de probabilidad discreta.
014 *
015 * El algoritmo usado es una variante del método de alias propuesta por Michael D. Vose,
016 * en su artículo "A Linear Algorithm For Generating Random Numbers With a Given Distribution".
017 * Esta variante tiene un tiempo de inicialización O(n) y tiempo de muestreo O(1) para todos
018 * los casos, lo que lo convierte en la más eficiente de su categoría.
019 * @author Alejandro González García
020 */
021final class DistribProbabilidadDiscreta {
022    /**
023     * El generador de números aleatorios de una distribución uniforme que usa esta distribución.
024     * Su calidad es decisiva para obtener unos resultados matemáticamente apropiados.
025     */
026    private final Random rng;
027    /**
028     * Variable de trabajo que contiene probabilidades intermedias.
029     */
030    private final double[] prob;
031    /**
032     * Variable de trabajo que contiene alias.
033     */
034    private final int[] alias;
035    /**
036     * El número de elementos para los que está definida la función masa de probabilidad.
037     */
038    private final int n;
039
040    /**
041     * Inicializa una nueva distribución de probabilidad discreta a partir de su función masa de
042     * probabilidad y un generador de números aleatorios.
043     * @param valoresMasaProb Los valores de la función masa de probabilidad. Los índices del array
044     * se interpretan como las preimágenes del valor correspondiente a su posición. Es decir, el valor
045     * 0 tendría una probabilidad valoresMasaProb[0] de salir; el 1, valoresMasaProb[1], y así
046     * sucesivamente. Aunque el algoritmo funciona igualmente con otro tipo de listas, es recomendable
047     * que implementen la interfaz RandomAccess, pues ello implica que los accesos a posiciones arbitrarias
048     * son de complejidad O(1), y la inicialización del objeto se completará más rápido.
049     * @param rng El generador de números aleatorios a usar para muestrear la distribución, cuando
050     * sea necesario.
051     */
052    public DistribProbabilidadDiscreta(final List<Double> valoresMasaProb, final Random rng) {
053        if (valoresMasaProb == null || valoresMasaProb.isEmpty()) {
054            throw new IllegalArgumentException("La función masa de probabilidad discreta no está definida para ningún valor.");
055        }
056
057        if (rng == null) {
058            throw new IllegalArgumentException("No se puede generar una distribución de probabilidad a partir de un RNG nulo.");
059        }
060        this.rng = rng;
061
062        // Inicialización de variables y estructuras de datos
063        this.n = valoresMasaProb.size();
064        this.alias = new int[n];
065        this.prob = new double[n];
066        final double equiprobabilidad = 1.0 / n;
067        final Deque<Integer> peque = new ArrayDeque<>(n);
068        final Deque<Integer> grande = new ArrayDeque<>(n);
069
070        // Inicializamos las colas de valores de la función de masa de probabilidad pequeños y grandes.
071        // De paso, comprobamos que la función masa de probabilidad sea coherente
072        double total = 0;
073        int i = 0;
074        for (double v : valoresMasaProb) {
075            if (v > equiprobabilidad) {
076                grande.push(i++);
077            } else {
078                peque.push(i++);
079            }
080
081            total += v;
082        }
083        // La función masa de probabilidad define el espacio muestral. Por tanto,
084        // incluye el suceso seguro, cuya probabilidad es 1. Permitimos algo de error
085        // al comprobar si la suma de las probabilidades es 1
086        if (total < 0.9999999 || total > 1.0000001) {
087            throw new IllegalArgumentException("La función masa de probabilidad no cumple su definición: la suma de todos sus valores no es 1.");
088        }
089
090        while (!peque.isEmpty() && !grande.isEmpty()) {
091            final int l = peque.pop();
092            final int g = grande.pop();
093
094            // Actualizamos valores según corresponda
095            prob[l] = n * valoresMasaProb.get(l);
096            alias[l] = g;
097
098            final double nuevaProb = valoresMasaProb.get(g) + (valoresMasaProb.get(l) - equiprobabilidad);
099            valoresMasaProb.set(g, nuevaProb);
100            if (nuevaProb > equiprobabilidad) {
101                grande.push(g);
102            } else {
103                peque.push(g);
104            }
105        }
106
107        // Consumir elementos restantes hasta agotarlos
108        while (!grande.isEmpty()) {
109            prob[grande.pop()] = 1;
110        }
111        while (!peque.isEmpty()) {
112            prob[peque.pop()] = 1;
113        }
114    }
115
116    /**
117     * Inicializa una nueva distribución de probabilidad discreta a partir de su función masa de
118     * probabilidad, usando como generador de números aleatorios la implementación de Random
119     * disponible en Java, basada en un generador lineal congruencial.
120     * @param valoresMasaProb Los valores de la función masa de probabilidad. Los índices del array
121     * se interpretan como las preimágenes del valor correspondiente a su posición. Es decir, el valor
122     * 0 tendría una probabilidad valoresMasaProb[0] de salir; el 1, valoresMasaProb[1], y así
123     * sucesivamente. Aunque el algoritmo funciona igualmente con otro tipo de listas, es recomendable
124     * que implementen la interfaz RandomAccess, pues ello implica que los accesos a posiciones arbitrarias
125     * son de complejidad O(1), y la inicialización del objeto se completará más rápido.
126     */
127    public DistribProbabilidadDiscreta(final List<Double> valoresMasaProb) {
128        this(valoresMasaProb, new Random());
129    }
130
131    /**
132     * Obtiene una muestra aleatoria de la distribución de probabilidad discreta que
133     * representa este objeto, de acuerdo con la probabilidad de que salga cada elemento.
134     * @return La devandicha muestra aleatoria.
135     */
136    public int next() {
137        final double u = rng.nextDouble() * n;
138        final int j = (int) u;
139        return u - j <= prob[j] ? j : alias[j];
140    }
