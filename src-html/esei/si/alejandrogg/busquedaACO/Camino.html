003package esei.si.alejandrogg.busquedaACO;
004
005import java.util.Iterator;
006import java.util.Spliterator;
007import java.util.Deque;
008import java.util.Queue;
009import java.util.NoSuchElementException;
010import java.util.concurrent.ConcurrentLinkedDeque;
011import java.util.concurrent.ConcurrentLinkedQueue;
012
013import esei.si.alejandrogg.busquedaACO.util.IteradorSoloLectura;
014import esei.si.alejandrogg.busquedaACO.util.SpliteratorSoloLectura;
015
016/**
017 * Modela un camino, que es una serie de carreteras (aristas) de un mapa (grafo).
018 * Una vez que se le han añadido las carreteras deseadas, un camino puede volverse inmutable,
019 * lo que facilita en gran medida la realización de operaciones eficientes con él entre
020 * varios hilos de ejecución. Un camino inmutable puede, además, compararse con otros caminos,
021 * según el orden natural de su distancia total.
022 * @author Alejandro González García
023 */
024public final class Camino implements Comparable<Camino>, Iterable<Carretera> {
025    /**
026     * El grafo de carreteras asociado a este camino, al que pertenecen las localidades y carreteras
027     * que formarán parte de este camino.
028     */
029    private final GrafoCarreteras grafoCarreteras;
030    /**
031     * La cola que contiene las carreteras por las que ha pasado este camino. Inicialmente está vacía.
032     */
033    private final Queue<Carretera> carreteras = new ConcurrentLinkedQueue<>();
034    /**
035     * La cola que contiene las localidades por las que ha pasado este camino, en orden FIFO.
036     * Inicialmente esta vacía, y su tamaño crece en múltiplos de dos.
037     */
038    private final Deque<Localidad> localidades = new ConcurrentLinkedDeque<>();
039    /**
040     * El {@link StringBuilder} usado internamente para ir construyendo la representación
041     * textual del camino según se va inicializando. De esta forma, sucesivas llamadas a
042     * {@link toString} no tienen que recomputar esta representación.
043     */
044    private StringBuilder sb = new StringBuilder();
045    /**
046     * La representación textual final del camino, para usar una vez esté inicializado.
047     */
048    private volatile String caminoComoCadena = null;
049    /**
050     * Toma el valor verdadero si y solo si el camino ya se ha construido totalmente; es decir,
051     * si el responsable de crearlo ha añadido todas las carreteras que forman parte de él, y
052     * no se espera que nadie más lo haga.
053     */
054    private volatile boolean construido = false;
055    /**
056     * La distancia total actual del camino ya construido, para evitar tener que computarla repetidamente.
057     */
058    private volatile int distanciaTotalActual = 0;
059    /**
060     * El código de dispersión actual del camino, para evitar tener que computarlo más de una vez.
061     */
062    private volatile int codigoDispersion = 1;
063    /**
064     * El número de localidades que forman parte del camino actualmente, para evitar tener que invocar
065     * el método size() en {@link localidades}, que tiene complejidad O(n).
066     */
067    private volatile int numLocalidades = 0;
068
069    /**
070     * Crea un nuevo camino, asociado a un determinado grafo de carreteras.
071     * @param grafoCarreteras El grafo de carreteras asociado al camino.
072     */
073    public Camino(final GrafoCarreteras grafoCarreteras) {
074        if (grafoCarreteras == null) {
075            throw new IllegalArgumentException("El grafo de carreteras asociado a un camino no puede ser nulo.");
076        }
077        this.grafoCarreteras = grafoCarreteras;
078    }
079
080    /**
081     * Añade una carretera al final de este camino, teniendo en cuenta que se ha tomado una carretera
082     * en un orden determinado, desde una localidad A hacia otra B. Esta operación tendrá éxito solo si
083     * el camino no se ha terminado de construir todavía, y asume que solo un hilo la emplea, al menos hasta
084     * que el camino se termine de construir. Para garantizar esta última asunción, basta con mantener
085     * la referencia a este objeto como variable local de un método hasta que se termine de construir.
086     * @param a La localidad desde la que se inicia el tramo del camino a añadir.
087     * @param b La localidad donde termina el tramo del camino a añadir.
088     * @throws IllegalArgumentException Si no hay una carretera que conecte ambas localidades.
089     * @throws IllegalStateException Si el camino ya se ha terminado de construir, pero se intenta añadir
090     * una nueva carretera a él.
091     */
092    public void addCarretera(final Localidad a, final Localidad b) {
093        final Carretera carretera = grafoCarreteras.getCarretera(a, b);
094        if (carretera == null) {
095            throw new IllegalArgumentException("No existe una carretera que conecte las localidades especificadas.");
096        }
097        if (construido) {
098            throw new IllegalStateException("No se puede añadir una carretera a un camino ya construido.");
099        }
100
101        // Añadir la carretera a la cola, actualizar el código de dispersión del camino,
102        // e incrementar la distancia total actual
103        carreteras.add(carretera);
104        codigoDispersion = 31 * codigoDispersion + carretera.hashCode(); // https://docs.oracle.com/javase/8/docs/api/java/util/List.html#hashCode--
105        distanciaTotalActual += carretera.getDistancia();
106
107        // Solo añadir la primera localidad si la lista está vacía, para evitar localidades
108        // repetidas
109        if (localidades.isEmpty()) {
110            localidades.add(a);
111            sb.append(a.getNombre()).append(", ");
112            ++numLocalidades;
113        }
114        localidades.add(b);
115        sb.append(b.getNombre()).append(", ");
116        ++numLocalidades;
117    }
118
119    /**
120     * Marca este camino como construido. Un camino ya construido no permite la adición de nuevas
121     * carreteras a él, volviéndose esencialmente de solo lectura, garantizando la coherencia de
122     * la información si varios hilos acceden al camino (pues todos ellos verán la última versión
123     * de los datos), y permitiendo operaciones de lectura sobre él. Este método está diseñado para
124     * ser llamado por el mismo que ha estado invocando {@link addCarretera}.
125     */
126    public void terminarConstruccion() {
127        // Eliminar coma sobrante
128        try {
129            sb.delete(sb.length() - 2, sb.length());
130        } catch (StringIndexOutOfBoundsException ignorar) {}
131
132        caminoComoCadena = sb.toString();
133        sb = null; // Ya no necesitamos seguir construyendo la cadena
134        construido = true;
135    }
136
137    /**
138     * Obtiene la última localidad que forma parte de este camino construido. En caso de que el
139     * camino no tenga todavía localidades, se devuelve el valor nulo.
140     * @return La devandicha localidad.
141     * @throws IllegalStateException Si el camino todavía no se ha terminado de construir.
142     */
143    public Localidad getUltimaLocalidad() {
144        if (construido) {
145            try {
146                return localidades.getLast();
147            } catch (NoSuchElementException | IndexOutOfBoundsException exc) {
148                return null;
149            }
150        } else {
151            throw new IllegalStateException("No se puede obtener la última localidad de un camino todavía no construido.");
152        }
153    }
154
155    /**
156     * Calcula la distancia total abarcada por este camino, si está ya construido.
157     * @implNote. La implementación de este método tiene complejidad O(1).
158     * @return La devandicha distancia.
159     * @throws IllegalStateException Si el camino todavía no se ha terminado de construir.
160     */
161    public int distanciaTotal() {
162        if (!construido) {
163            throw new IllegalStateException("No se puede obtener la distancia total de un camino todavía no construido.");
164        }
165
166        return distanciaTotalActual;
167    }
168
169    /**
170     * {@inheritDoc}
171     * @implNote. La implementación de este método tiene complejidad O(1).
172     * @throws IllegalStateException Si algún camino implicado no está construido.
173     */
174    @Override
175    public int compareTo(final Camino otro) {
176        if (otro == null) {
177            throw new NullPointerException("No se puede comparar un camino con otro nulo.");
178        }
179
180        return distanciaTotal() - otro.distanciaTotal();
181    }
182
183    /**
184     * {@inheritDoc}
185     * @implNote. La implementación de este método tiene complejidad O(n) en el peor caso,
186     * siendo n el número de localidades por las que pasa el camino, por lo que solo se
187     * recomienda usar este método si es estrictamente necesario. No obstante, si es obvio
188     * que los caminos no pueden ser iguales por tener longitudes diferentes, la complejidad
189     * se reduce a O(1).
190     * @throws IllegalStateException Si algún camino implicado no está construido.
191     */
192    @Override
193    public boolean equals(final Object otro) {
194        boolean toret = otro instanceof Camino;
195        Camino otroCamino;
196
197        if (toret) {
198            otroCamino = (Camino) otro;
199
200            if (!construido || !otroCamino.construido) {
201                throw new IllegalStateException("No se puede comparar la igualdad de dos caminos cuando al menos uno no está construido.");
202            }
203
204            // Si no tienen el mismo tamaño, ni de lejos son iguales
205            toret = numLocalidades == otroCamino.numLocalidades;
206            if (toret) {
207                // TODO: quizás optimizar esto con Spliterator
208                final Iterator<Localidad> iter1 = localidades.iterator();
209                final Iterator<Localidad> iter2 = localidades.iterator();
210
211                while (toret && iter1.hasNext()) {
212                    final Localidad l1 = iter1.next();
213                    final Localidad l2 = iter2.next();
214                    toret = l1.equals(l2);
215                }
216            }
217        }
218
219        return toret;
220    }
221
222    /**
223     * {@inheritDoc}
224     * @implNote. La implementación de este método tiene complejidad O(1).
225     * @throws IllegalStateException Si el camino no está construido.
226     */
227    @Override
228    public int hashCode() {
229        if (!construido) {
230            throw new IllegalStateException("No se puede obtener la clave de dispersión de un camino no construido.");
231        }
232
233        return codigoDispersion;
234    }
235
236    /**
237     * {@inheritDoc}
238     * @throws IllegalStateException Si el camino no está construido.
239     */
240    @Override
241    public Iterator<Carretera> iterator() {
242        if (!construido) {
243            throw new IllegalStateException("No se puede obtener un iterador sobre un camino no construido.");
244        }
245
246        return new IteradorSoloLectura<>(carreteras.iterator());
247    }
248
249    /**
250     * {@inheritDoc}
251     * @throws IllegalStateException Si el camino no está construido.
252     */
253    @Override
254    public Spliterator<Carretera> spliterator() {
255        if (!construido) {
256            throw new IllegalStateException("No se puede obtener un iterador sobre un camino no construido.");
257        }
258
259        return new SpliteratorSoloLectura<>(carreteras.spliterator());
260    }
261
262    /**
263     * {@inheritDoc}
264     * @implNote. La implementación de este método tiene complejidad O(1).
265     */
266    @Override
267    public String toString() {
268        String toret = "Inicializando...";
269
270        if (construido) {
271            toret = caminoComoCadena;
272        }
273
274        return toret;
275    }
