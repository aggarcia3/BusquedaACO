003package esei.si.alejandrogg.busquedaACO;
004
005import java.util.logging.Level;
006import java.util.Map;
007import java.util.concurrent.ConcurrentHashMap;
008
009/**
010 * Fabricación pura que se responsabiliza de mantener el control sobre
011 * qué hormiga se encuentra en qué localidad en todo momento.
012 * @author Alejandro González García
013 */
014final class LocalizadorHormigas {
015    /**
016     * El mundo asociado a este localizador de hormigas, sobre el que brinda sus servicios.
017     */
018    private final Mundo mundo;
019    /**
020     * El mapa que relaciona a cada hormiga con la localidad en la que se encuentra.
021     */
022    private final Map<String, Localidad> ubicacionHormiga = new ConcurrentHashMap<>();
023
024    /**
025     * Crea un nuevo localizador de hormigas, asociado con un determinado mundo.
026     * @param mundo El mundo asociado con el nuevo localizador.
027     * @throws IllegalArgumentException Si el parámetro es nulo.
028     */
029    LocalizadorHormigas(final Mundo mundo) {
030        if (mundo == null) {
031            throw new IllegalArgumentException("No se puede crear un localizador de hormigas que localice hormigas de un mundo nulo.");
032        }
033        this.mundo = mundo;
034    }
035
036    /**
037     * Registra el movimiento de una hormiga de la localidad en la que esté (si está en alguna)
038     * a otra localidad. Este método es seguro en caso de ser llamado por varios hilos de manera concurrente.
039     * @param hormiga La hormiga cuyo movimiento registrar.
040     * @param localidad La localidad a la que pasa a estar la hormiga.
041     * @throws IllegalArgumentException Si algún parámetro es nulo.
042     */
043    void mover(final String hormiga, final Localidad localidad) {
044        if (hormiga == null) {
045            throw new IllegalArgumentException("No se puede colocar una hormiga de nombre nulo en una localidad.");
046        }
047        if (localidad == null) {
048            throw new IllegalArgumentException("No se puede colocar una hormiga en una localidad nula.");
049        }
050
051        // Javadoc del método compute: "The entire method invocation is performed atomically".
052        // De esta forma, nos ahorramos sincronización externa para hacer atómicas operaciones no atómicas:
053        // comprobar la localidad anterior en el mapa, y si es distinta de la nueva, borrar y añadir una hormiga
054        // en las localidades correspondientes
055        ubicacionHormiga.compute(hormiga, (final String ignorada, final Localidad locAnterior) -> {
056            if (locAnterior != localidad) {
057                // Eliminar la hormiga de la localidad anterior
058                if (locAnterior != null) {
059                    locAnterior.borrarHormiga();
060                    mundo.getLogger().log(Level.FINE, "La hormiga " + hormiga + " se desplaza de " + locAnterior + " a " + localidad + ".");
061                }
062
063                // Añadirla a la nueva localidad
064                localidad.addHormiga();
065            }
066
067            return localidad;
068        });
069    }
070
071    /**
072     * Registra el movimiento de todas las hormigas de la localidad en la que estén a otra localidad.
073     * Este método no es seguro en caso de ser llamado por varios hilos de manera concurrente, pues
074     * no realiza sus acciones de manera atómica, y sería posible que otros hilos viesen resultados
075     * parciales.
076     * @param localidad La localidad a la que pasan a estar las hormigas.
077     * @throws IllegalArgumentException Si algún parámetro es nulo.
078     */
079    void moverTodasALocalidad(final Localidad localidad) {
080        if (localidad == null) {
081            throw new IllegalArgumentException("No se pueden colocar hormigas en una localidad nula.");
082        }
083
084        // Las expresiones lambda fueron un buen añadido a Java 8. ¿Acaso no es elegante, conciso y expresivo lo siguiente? :)
085        ubicacionHormiga.replaceAll((final String hormiga, final Localidad locAnterior) -> localidad);
086    }
087
088    /**
089     * Provoca la desaparición de una hormiga del mapa, eliminando su ubicación registrada.
090     * @param hormiga La hormiga a borrar del mapa.
091     * @throws IllegalArgumentException Si el parámetro es nulo.
092     */
093    void desaparecer(final String hormiga) {
094        if (hormiga == null) {
095            throw new IllegalArgumentException("No se puede hacer desaparecer una hormiga nula.");
096        }
097
098        ubicacionHormiga.remove(hormiga).borrarHormiga();
099    }
100
101    /**
102     * Obtiene la localidad en la que se encuentra actualmente una hormiga.
103     * @param hormiga La hormiga cuya localidad actual obtener.
104     * @return La localidad en la que está la hormiga. Si la hormiga no se encuentra
105     * en ninguna localidad, lo cual puede pasar porque todavía no se ha registrado
106     * como tal, se devuelve el valor nulo.
107     * @throws IllegalArgumentException Si el parámetro es nulo.
108     */
109    Localidad ubicacion(final String hormiga) {
110        if (hormiga == null) {
111            throw new IllegalArgumentException("No se puede obtener la ubicación de una hormiga nula.");
112        }
113
114        return ubicacionHormiga.get(hormiga);
115    }
116}
