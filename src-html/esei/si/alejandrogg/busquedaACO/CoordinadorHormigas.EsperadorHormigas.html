005import java.util.Set;
006import java.util.HashSet;
007import java.util.logging.Level;
008import java.util.function.Consumer;
009import java.util.concurrent.TimeUnit;
010import java.util.concurrent.locks.Condition;
011import java.util.concurrent.locks.Lock;
012import java.util.concurrent.locks.ReentrantLock;
013
014import static java.lang.System.currentTimeMillis;
015
016/**
017 * Esta clase provee servicios y toma responsabilidades relacionadas con la correcta
018 * sincronización entre agentes de hormigas, para garantizar la corrección y consistencia
019 * de la simulación independientemente de la velocidad de ejecución de las acciones
020 * individuales de los agentes. Es una fabricación pura.
021 *
022 * CoordinadorHormigas tiene una relación relativamente estrecha con {@link Mundo}, que se refleja
023 * en un contrato relativamente estricto, con dos términos: esta clase no debe invocar métodos
024 * de {@link Mundo} sin asegurarse previamente de que se han inicializado los datos que usan.
025 * Destacablemente, invocar los métodos getLogger() y getNHormigas() es, por construcción de la
026 * clase {@link Mundo}, siempre seguro. Por otro lado, el {@link Mundo} se debe de comprometer a no propagar
027 * instancias de CoordinadorHormigas antes de que complete su propia inicialización, lo cual actualmente
028 * hace mediante una espera por una condición en su método get(). Este último término es necesario para
029 * garantizar el correcto funcionamiento de la espera inicial por las hormigas.
030 *
031 * @see Mundo
032 * @author Alejandro González García
033 */
034public final class CoordinadorHormigas {
035    /**
036     * El mundo sobre el que este coordinador presta sus servicios.
037     */
038    private final Mundo mundo;
039    /**
040     * Candado que se utiliza para sincronizar varios hilos de ejecución
041     * cuando se inicia la espera por hormigas.
042     */
043    private final Lock candadoInicioEspera = new ReentrantLock();
044    /**
045     * Condición de inicio de espera por hormigas del SMA.
046     */
047    private final Condition condicionInicioEspera = candadoInicioEspera.newCondition();
048    /**
049     * Candado (lock) que se utiliza para sincronizar varios hilos de ejecución
050     * cuando llegan hormigas.
051     */
052    private final Lock candadoLlegadaHormigas = new ReentrantLock();
053    /**
054     * Condición de llegada de hormigas al SMA.
055     */
056    private final Condition condicionLlegada = candadoLlegadaHormigas.newCondition();
057    /**
058     * El candado que utilizamos para establecer relaciones pasa-antes necesarias
059     * para la atomicidad y visibilidad de la llegada de las respuestas de las hormigas.
060     * Aunque la clase Phaser de Java provee funcionalidades similares, por simplicidad
061     * es deseable no tener que gestionar también la visibilidad de la referencia al Phaser
062     * creado, de forma que se pueda saber en todo momento cuántas hormigas quedan por
063     * responder, y si se está esperando por la respuesta de hormigas, con tan solo consultar
064     * el valor de un entero. Además, así también podríamos reducir el número de candados
065     * que los hilos tienen que adquirir para hacer su cometido.
066     */
067    private final Lock candadoEsperaRespuesta = new ReentrantLock();
068    /**
069     * La condición que la última hormiga en responder indicará que ha ocurrido, para notificar
070     * a quien espera por la respuesta de todas las hormigas de que su espera se ha completado.
071     */
072    private final Condition condicionEsperaRespuesta = candadoEsperaRespuesta.newCondition();
073
074    /**
075     * Tarea cuyo código se ejecutará cuando llegue una hormiga al sistema.
076     */
077    private final Consumer<String> tareaLlegadaHormiga;
078    /**
079     * Tarea cuyo código se ejecutará cuando una hormiga deje de formar parte del
080     * sistema.
081     */
082    private final Consumer<String> tareaMuerteHormiga;
083    /**
084     * Indica si ha llegado recientemente una hormiga al sistema que todavía
085     * no fue atendida por esta clase. Los accesoa a esta variable, tanto de lectura
086     * como de escritura, deben de sincronizarse con el monitor de {@link condicionLlegada}.
087     * Se usa para evitar que tengan influencia en el sistema los "spurious wakeups".
088     */
089    private boolean hormigaRecienLlegada = false;
090    /**
091     * Toma el valor verdadero si y solo si nos encontramos esperando por la llegada
092     * de hormigas actualmente.
093     */
094    private volatile boolean esperandoHormigas = false;
095    /**
096     * El conjunto de hormigas que enviaron una respuesta durante la espera actual por respuestas.
097     * Se usa para no contar más de una vez respuestas múltiples. Los accesos a este TAD deben
098     * de sincronizarse con el candado {@link candadoEsperaRespuesta}.
099     */
100    private final Set<String> hormigasQueRespondieron = new HashSet<>();
101    /**
102     * El número de hormigas que quedan por responder en el momento presente.
103     * Este número es negativo si no se está esperando a que las hormigas respondan.
104     * Los accesos a esta variable deben de sincronizarse con el candado {@link candadoEsperaRespuesta}.
105     */
106    private int hormigasEsperandoRespuesta = Integer.MIN_VALUE;
107
108    /**
109     * Crea un nuevo coordinador de hormigas, con sus tareas asociadas a ejecutar
110     * de modo asíncrono cuando ocurran ciertos eventos.
111     * @param mundo El mundo cuyas hormigas este objeto coordinará. Como contrato general, esta clase no debe
112     * invocar métodos en este objeto sin asegurarse previamente de que se han inicializado los datos que usan.
113     * Destacablemente, invocar los métodos getLogger() y getNHormigas() es, por construcción de la clase Mundo,
114     * siempre seguro.
115     * @param tareaLlegadaHormiga La tarea cuyo código se ejecutará cuando llegue una hormiga al sistema.
116     * @param tareaMuerteHormiga La tarea cuyo código se ejecutará cuando una hormiga deje de formar parte del SMA.
117     * @throws IllegalArgumentException Si algún parámetro es nulo.
118     */
119    CoordinadorHormigas(final Mundo mundo, final Consumer<String> tareaLlegadaHormiga, final Consumer<String> tareaMuerteHormiga) {
120        if (mundo == null) {
121            throw new IllegalArgumentException("El mundo asociado a un coordinador de hormigas no puede ser nulo.");
122        }
123        if (tareaLlegadaHormiga == null) {
124            throw new IllegalArgumentException("La tarea a realizar cuando llegue una hormiga no puede ser nula.");
125        }
126        if (tareaMuerteHormiga == null) {
127            throw new IllegalArgumentException("La tarea a realizar cuando una hormiga deje de formar parte del sistema no puede ser nula.");
128        }
129        this.mundo = mundo;
130        this.tareaLlegadaHormiga = tareaLlegadaHormiga;
131        this.tareaMuerteHormiga = tareaMuerteHormiga;
132    }
133
134    /**
135     * Implementa una tarea que espera a que aparezcan hormigas en el sistema, esperando
136     * a lo sumo un tiempo configurable desde la llegada de la última hormiga o el inicio
137     * de la espera.
138     * @author Alejandro González García
139     */
140    private class EsperadorHormigas implements Runnable {
141        /**
142         * Los milisegundos a esperar a que llegue una nueva hormiga, contados
143         * desde la llegada de la última hormiga o desde el inicio de la espera.
144         */
145        private final int msEspera;
146        /**
147         * La tarea cuyo método run() se ejecutará cuando se termine de realizar
148         * la espera.
149         */
150        private final Runnable tareaFinEspera;
151        /**
152         * Toma el valor verdadero si el método run() ha comenzado su ejecución,
153         * y está listo para procesar la llegada de hormigas.
154         */
155        private volatile boolean hiloIniciado = false;
156
157        /**
158         * Crea un nuevo esperador de hormigas, con un determinado tiempo de espera máximo.
159         * @param segundosEspera El devandicho tiempo de espera.
160         * @param tareaFinEspera La tarea a ejecutar cuando finalice la espera.
161         */
162        EsperadorHormigas(final short segundosEspera, final Runnable tareaFinEspera) {
163            if (segundosEspera <= 0) {
164                throw new IllegalArgumentException("No se puede esperar una cantidad de tiempo negativa o nula a que lleguen hormigas.");
165            }
166            if (tareaFinEspera == null) {
167                throw new IllegalArgumentException("La tarea a ejecutar cuando se termine de esperar por la creación de hormigas no puede ser nula.");
168            }
169
170            this.msEspera = ((int) segundosEspera) * 1000;
171            this.tareaFinEspera = tareaFinEspera;
172        }
173
174        /**
175         * Devuelve verdadero si y solo si el objeto está listo para procesar
176         * la llegada de hormigas.
177         * @return El descrito valor de retorno.
178         */
179        boolean hiloIniciado() {
180            return hiloIniciado;
181        }
182
183        @Override
184        public void run() {
185            final long inicioEspera = currentTimeMillis();
186            final Thread hiloActual = Thread.currentThread();
187            boolean tiempoExpirado = false;
188
189            // Mientras no expire el tiempo desde que llegó la última hormiga, esperar
190            while (!tiempoExpirado) {
191                candadoLlegadaHormigas.lock();
192                try {
193                    final long finalEspera = currentTimeMillis() + msEspera;
194
195                    // Avisar a otros hilos de que nos hemos iniciado, y estamos listos para
196                    // atender la llegada de hormigas
197                    if (!hiloIniciado) {
198                        candadoInicioEspera.lock();
199                        try {
200                            condicionInicioEspera.signal();
201                            hiloIniciado = true;
202                        } finally {
203                            candadoInicioEspera.unlock();
204                        }
205                    }
206
207                    while (!hormigaRecienLlegada && !tiempoExpirado) {
208                        // Esperar msEspera ms a que la llegada de una hormiga active la condición
209                        // de espera. Si no lo hace ninguna hormiga en ese plazo, entonces el tiempo
210                        // expira y dejamos de esperar
211                        final long msEspera = finalEspera - currentTimeMillis();
212                        if (msEspera > 0) {
213                            try {
214                                condicionLlegada.await(msEspera, TimeUnit.MILLISECONDS);
215                            } catch (InterruptedException exc) {
216                                // Informar de que hemos recibido la petición de interrupción,
217                                // pero no hacerle caso
218                                mundo.getLogger().log(Level.INFO, "Todavía se espera por la siguiente hormiga...");
219                            }
220                        }
221
222                        // Si hemos esperado tanto que hemos llegado al momento límite, entonces es
223                        // que ha expirado el tiempo
224                        if (msEspera <= 0) {
225                            mundo.getLogger().log(Level.INFO, "Se ha esperado por " + mundo.getNHormigas() + " hormiga/s durante " + ((currentTimeMillis() - inicioEspera) / 1000.0) + " s.");
226                            tiempoExpirado = true;
227                        }
228                    }
229
230                    // Si tras la espera ha llegado una hormiga, dejar de considerarla como recién llegada
231                    if (hormigaRecienLlegada) {
232                        hormigaRecienLlegada = false;
233                    }
234                } finally {
235                    candadoLlegadaHormigas.unlock();
236                }
237            }
238
239            // Hemos completado la espera, así que ejecutar la tarea pertinente
240            CoordinadorHormigas.this.esperandoHormigas = false;
241            tareaFinEspera.run();
242        }
243    }
244
245    /**
246     * Espera a que lleguen (se creen) hormigas iniciales en el SMA, deteniendo la espera si pasa un
247     * tiempo desde la llegada de la última hormiga, o pasa ese tiempo sin que llegue ninguna hormiga.
248     * Cuando termine la espera se ejecutará una tarea, que se pasa como parámetro. La espera es finita y
249     * terminará en algún momento, si llega un número finito de hormigas al sistema. Cuando este
250     * método devuelve el control a quien lo llama, se garantiza que la espera a la llegada de hormigas se
251     * haya puesto en marcha, y el coordinador se encuentra, a todos los efectos, listo para atender tales
252     * eventos.
253     * @param tareaFinEspera La tarea a ejecutar cuando termine la espera.
254     * @param segundosEspera El tiempo máximo a esperar desde la llegada de la última hormiga o el inicio
255     * de la espera.
256     * @throws IllegalArgumentException Si segundosEspera es menor o igual que 0, o tareaFinEspera es nula.
257     */
258    void esperarLlegadaHormigas(final Runnable tareaFinEspera, final short segundosEspera) {
259        // Estamos esperando por la llegada de hormigas sí o sí
260        esperandoHormigas = true;
261
262        // Avisamos al usuario de la intención de espera
263        mundo.getLogger().log(Level.INFO, "Esperando a que lleguen las hormigas al sistema multiagente...");
264
265        // Creamos el hilo para la espera, y le decimos a la JVM que lo inicie
266        final EsperadorHormigas eh = new EsperadorHormigas(segundosEspera, tareaFinEspera);
267        new Thread(eh).start();
268
269        // No sabemos cuándo se va a iniciar el hilo (la JVM lo hará, en coordinación con el
270        // SO, en algún momento futuro desconocido). Por tanto, debemos de esperar por él para
271        // garantizar una postcondición de salida de este método
272        candadoInicioEspera.lock();
273        try {
274            while (!eh.hiloIniciado()) {
275                try {
276                    final boolean tiempoNoExpirado = condicionInicioEspera.await(10, TimeUnit.SECONDS);
277                    if (!tiempoNoExpirado) {
278                        mundo.getLogger().log(Level.WARNING, "El hilo de espera de llegada de hormigas no se ha iniciado en 10 segundos. Si la espera tarda demasiado, por favor reinicie el sistema.");
279                    }
280                } catch (InterruptedException exc) {
281                    mundo.getLogger().log(Level.INFO, "Esperando a que se inicie el hilo de espera de llegada de hormigas...");
282                }
283            }
284        } finally {
285            candadoInicioEspera.unlock();
286        }
287    }
288
289    /**
290     * Espera a que todas las hormigas de las que se tiene constancia en el SMA terminen
291     * su reacción a la percepción de paso de tiempo. Este método no devuelve el control a
292     * quien lo llama hasta que todas las hormigas han reaccionado, o bien ha pasado un tiempo
293     * máximo.
294     * @param tareaNotificacionRespuesta La tarea que se responsabiliza de enviar la percepción de
295     * que el entorno está listo para esperar por la respuesta de las hormigas. Se ejecutará en
296     * el mismo hilo que ha llamado a este método.
297     * @param segundosEspera Los segundos a esperar como máximo a que todas las hormigas
298     * respondan. Un valor menor o igual que 0 significa esperar indefinidamente.
299     * @throws IllegalStateException Si se vuelve a intentar esperar por la respuesta de hormigas
300     * mientras tal espera está teniendo lugar.
301     * @throws IllegalArgumentException Si el parámetro tareaNotificacionRespuesta es nulo.
302     */
303    void esperarRespuestaHormigas(final Runnable tareaNotificacionRespuesta, final short segundosEspera) {
304        if (tareaNotificacionRespuesta == null) {
305            throw new IllegalArgumentException("La tarea de notificación de que el SMA está listo para esperar por la respuesta de hormigas no puede ser nula.");
306        }
307
308        candadoEsperaRespuesta.lock();
309        try {
310            // Comprobar que no estamos ya esperando por la respuesta de hormigas
311            if (hormigasEsperandoRespuesta >= 0) {
312                throw new IllegalStateException("No se puede esperar por la respuesta de las hormigas mientras tal espera está ya teniendo lugar.");
313            }
314
315            // Avisar al usuario del inicio de la espera
316            mundo.getLogger().log(Level.INFO, "Esperando a que todas las hormigas ejecuten su respuesta...");
317
318            // Inicialmente, esperaremos por la respuesta de tantas hormigas como haya.
319            // Las acciones de las hormigas (llegada, muerte y respuesta) influirán en el
320            // número de hormigas por las que se espera la respuesta
321            hormigasEsperandoRespuesta = mundo.getNHormigas();
322
323            // Solo tiene sentido esperar si hay al menos una hormiga
324            final long inicioEspera = currentTimeMillis();
325            if (hormigasEsperandoRespuesta > 0) {
326                // Notificar a los agentes hormigas que estamos listos para atender sus respuestas.
327                // Si hiciésemos esta notificación en otro momento, fuera de esta sección crítica, no garantizaríamos
328                // que viesen el inicio de la espera
329                tareaNotificacionRespuesta.run();
330
331                // Cuando todas las hormigas hayan terminado, terminar la espera
332                if (segundosEspera > 0) {
333                    long inicioIteracionEspera;
334                    long tiempoRestante = segundosEspera * 1000;
335
336                    while (hormigasEsperandoRespuesta > 0 && tiempoRestante > 0) {
337                        inicioIteracionEspera = currentTimeMillis();
338
339                        try {
340                            condicionEsperaRespuesta.await(tiempoRestante, TimeUnit.MILLISECONDS);
341                        } catch (InterruptedException ignorar) {}
342
343                        // Descontar el tiempo transcurrido esperando del tiempo restante
344                        tiempoRestante -= currentTimeMillis() - inicioIteracionEspera;
345                    }
346                } else {
347                    try {
348                        while (hormigasEsperandoRespuesta > 0) {
349                            condicionEsperaRespuesta.await();
350                        }
351                    } catch (InterruptedException exc) {
352                        mundo.getLogger().log(Level.WARNING, "Se ha interrumpido la espera por la respuesta de las hormigas antes de que se completase. No se garantiza que se hayan recibido todas las respuestas posibles, y el resultado podría ser inconsistente.");
353                    }
354                }
355            } else {
356                mundo.getLogger().log(Level.INFO, "No hay hormigas por las que esperar.");
357            }
358
359            mundo.getLogger().log(Level.INFO, "Espera por respuesta de " + hormigasQueRespondieron.size() + " hormiga/s completada en " + ((currentTimeMillis() - inicioEspera) / 1000.0) + " s.");
360
361            // Ya no estamos esperando por la respuesta de hormigas
362            hormigasEsperandoRespuesta = Integer.MIN_VALUE;
363            hormigasQueRespondieron.clear();
364        } finally {
365            candadoEsperaRespuesta.unlock();
366        }
367    }

369    /**
370     * Comprueba si se está esperando actualmente por la llegada de hormigas al sistema.
371     * @return El valor de verdad de la proposición descrita.
372     */
373    boolean esperandoLlegadaHormigas() {
374        return esperandoHormigas;
375    }

377    /**
378     * Maneja el evento de llegada de una hormiga al SMA, enviando las notificaciones
379     * pertinentes.
380     * @param nombre La hormiga que llega.
381     */
382    public void llegadaHormiga(final String nombre) {
383        // Notificar a la espera por hormigas, si está teniendo lugar tal espera
384        // (asumimos que la inicialización del entorno se completa antes de que la creación
385        // asíncrona de agentes obtenga una referencia a este coordinador)
386        if (esperandoHormigas) {
387            candadoLlegadaHormigas.lock();
388            try {
389                condicionLlegada.signal();
390                hormigaRecienLlegada = true;
391            } finally {
392                candadoLlegadaHormigas.unlock();
393            }
394        }
395
396        // Si hay una espera por la respuesta de hormigas en curso, registrarnos
397        // como participante
398        candadoEsperaRespuesta.lock();
399        try {
400            if (hormigasEsperandoRespuesta >= 0) {
401                ++hormigasEsperandoRespuesta;
402            }
403        } finally {
404            candadoEsperaRespuesta.unlock();
405        }
406
407        // Realizar la tarea de llegada de hormiga
408        tareaLlegadaHormiga.accept(nombre);
409    }
410
411    /**
412     * Maneja el evento de abandono de una hormiga del SMA, enviando las notificaciones
413     * pertinentes.
414     * @param nombre La hormiga que deja de formar parte del SMA.
415     */
416    public void muerteHormiga(final String nombre) {
417        // Si hay una espera por la respuesta de hormigas en curso, cancelar
418        // nuestra participación en la respuesta
419        candadoEsperaRespuesta.lock();
420        try {
421            if (hormigasEsperandoRespuesta > 0) {
422                // Avisar a quien espera por la respuesta de todas las hormigas si esta fue la última
423                if (--hormigasEsperandoRespuesta == 0) {
424                    condicionEsperaRespuesta.signal();
425                }
426            }
427        } finally {
428            candadoEsperaRespuesta.unlock();
429        }
430
431        // Realizar la tarea de muerte de hormiga
432        tareaMuerteHormiga.accept(nombre);
433    }
434
435    /**
436     * Maneja el evento de respuesta de una hormiga a la percepción de paso del tiempo,
437     * enviando las notificaciones pertinentes.
438     * @param nombre El nombre de la hormiga que responde a la percepción.
439     */
440    public void respuestaHormiga(final String nombre) {
441        // En principio, solo tiene sentido que las hormigas envíen respuestas si estamos esperando
442        // por hormigas
443        candadoEsperaRespuesta.lock();
444        try {
445            if (hormigasEsperandoRespuesta > 0) {
446                // Solo contar una respuesta por espera
447                if (!hormigasQueRespondieron.contains(nombre)) {
448                    hormigasQueRespondieron.add(nombre);
449
450                    // Avisar a quien espera por la respuesta de todas las hormigas si esta fue la última
451                    if (--hormigasEsperandoRespuesta == 0) {
452                        condicionEsperaRespuesta.signal();
453                    }
454                }
455            } else {
456                // Esto no debería de ocurrir: se espera recibir la respuesta solo si estamos esperando la respuesta de al menos una hormiga
457                mundo.getLogger().log(Level.WARNING, "La hormiga " + nombre + " ha respondido, pero el entorno no esperaba que lo hiciese en este momento. Esto puede ser síntoma de algún problema.");
458            }
459        } finally {
460            candadoEsperaRespuesta.unlock();
461        }
462    }
