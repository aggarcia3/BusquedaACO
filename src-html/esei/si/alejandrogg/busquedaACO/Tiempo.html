003package esei.si.alejandrogg.busquedaACO;
004
005import java.util.logging.Level;
006import java.util.concurrent.Callable;
007
008import static java.lang.System.currentTimeMillis;
009
010/**
011 * Una fabricación pura que se responsabiliza del correcto avance del tiempo
012 * discreto del sistema, en coordinación con el {@link Mundo}. Se ejecuta en
013 * otro hilo. El comportamiento señalizador del siguiente instante de tiempo
014 * es completamente personalizable. Proporciona más control sobre la variabilidad
015 * del periodo de avance que la interfaz ScheduledExecutorService de Java.
016 * @author Alejandro González García
017 */
018final class Tiempo implements Runnable {
019    /**
020     * El mundo sobre el que esta clase brindará sus servicios.
021     */
022    private final Mundo mundo;
023    /**
024     * El hilo encargado de realizar el avance periódico del tiempo.
025     */
026    private final Thread hilo;
027    /**
028     * Contiene el código a ejecutar en su método {@link Callable#call} para notificar
029     * el paso del tiempo.
030     */
031    private final Callable<Boolean> tick;
032    /**
033     * Los milisegundos que pasan entre dos instantes de tiempo discretos.
034     */
035    private volatile int msPeriodo;
036    /**
037     * Toma el valor verdadero si se ha iniciado el avance del tiempo discreto,
038     * falso en caso contrario.
039     */
040    private volatile boolean iniciado = false;
041
042    /**
043     * Crea un nuevo objeto responsable de ejecutar el avance periódico del tiempo.
044     * @param mundo El mundo (entorno) sobre el que se pasará el tiempo.
045     * @param tick El código a ejecutar para notificar el paso del tiempo. Si devuelve
046     * verdadero, se detendrá la ejecución del avance periódico del tiempo.
047     * @param msPeriodo Los milisegundos que deben transcurrir entre dos instantes
048     * de tiempo consecutivos.
049     * @throws IllegalArgumentException Si msPeriodo es estrictamente menor que 0,
050     * tick es nulo o mundo es nulo.
051     */
052    Tiempo(final Mundo mundo, final Callable<Boolean> tick, final int msPeriodo) {
053        if (mundo == null) {
054            throw new IllegalArgumentException("El tiempo de un mundo nulo no puede avanzar.");
055        }
056        if (tick == null) {
057            throw new IllegalArgumentException("No tiene sentido que la tarea a ejecutar para notificar el paso del tiempo sea nula.");
058        }
059
060        this.mundo = mundo;
061        this.tick = tick;
062        this.hilo = new Thread(this, "Tiempo");
063        hilo.setDaemon(true);
064
065        setPeriodo(msPeriodo);
066    }
067
068    /**
069     * Cambia el periodo que transcurre entre dos instantes de tiempo consecutivos. La acción
070     * de este método tendrá efecto tras la próxima ejecución de {@link tick}.
071     * @param msPeriodo El nuevo periodo a establecer.
072     * @throws IllegalArgumentException Si msPeriodo es estrictamente menor que 0.
073     */
074    void setPeriodo(final int msPeriodo) {
075        if (msPeriodo < 0) {
076            throw new IllegalArgumentException("No puede pasar un tiempo negativo entre dos instantes de tiempo discretos.");
077        }
078        this.msPeriodo = msPeriodo;
079    }
080
081    /**
082     * Comienza el avance periódico del tiempo del SMA, en otro hilo de ejecución. Si el hilo actual ya
083     * ha comenzado el avance del tiempo del SMA, este método no tiene efecto. Este método no es seguro
084     * para ser ejecutado de manera concurrente. El avance periódico del tiempo no se puede reiniciar
085     * una vez parado.
086     * @return La instancia del objeto sobre el que se llama a este método.
087     */
088    Tiempo iniciar() {
089        if (!iniciado) {
090            hilo.start();
091            iniciado = true;
092        }
093        return this;
094    }
095
096    /**
097     * Detiene de manera ordenada el avance del tiempo. Tras la ejecución de este método
098     * se ha hecho un mejor intento para detener la ejecución de {@link tick}.
099     */
100    void parar() {
101        if (iniciado) {
102            // Avisar al hilo para que se detenga
103            hilo.interrupt();
104
105            // Esperar hasta 10 segundos para que acate la orden
106            try {
107                hilo.join(10000);
108            } catch (InterruptedException ignorar) {}
109
110            // Si el hilo sigue ejecutándose pese a todo, avisar de ello
111            // (como es un hilo de fondo, daemon, no debería de afectar al cierre de la JVM)
112            if (!hilo.getState().equals(Thread.State.TERMINATED)) {
113                mundo.getLogger().log(Level.WARNING, "Se intentó parar el hilo responsable de avanzar el tiempo en el SMA, pero no respondió en 10 segundos. Continuando de todas formas.");
114            }
115        }
116    }
117
118    @Override
119    public void run() {
120        final Thread esteHilo = Thread.currentThread();
121        boolean seDeseaParar = false;
122
123        while (!seDeseaParar && !esteHilo.interrupted()) {
124            // Avanzamos al siguiente instante, observando lo que hemos tardado en procesar
125            // el tick
126            mundo.getLogger().log(Level.INFO, "Comienza la simulación del siguiente instante de tiempo.");
127            final long timestampInicioTick = currentTimeMillis();
128            try {
129                seDeseaParar = tick.call();
130            } catch (Exception exc) {
131                mundo.getLogger().log(Level.WARNING, "Ha ocurrido un error no esperado mientras se ejecutaba la tarea de avance de tiempo del entorno. Detalles: ", exc);
132            }
133            final long timestampFinTick = currentTimeMillis();
134            mundo.getLogger().log(Level.INFO, "Finaliza la simulación del siguiente instante de tiempo.");
135
136            // Dormirnos si nos sobra tiempo hasta el siguiente tick, y no queremos pararnos
137            if (!seDeseaParar && timestampFinTick - timestampInicioTick < msPeriodo) {
138                try {
139                    esteHilo.sleep(msPeriodo - (timestampFinTick - timestampInicioTick));
140                } catch (InterruptedException exc) {
141                    // Restaurar bandera de interrupción, para salir del bucle
142                    esteHilo.interrupt();
143                } catch (IllegalArgumentException ignorar) {
144                    // No interrumpir la ejecución aunque el resultado de la operación
145                    // sea negativo (overflow). Preferimos hacer el siguiente tick a destiempo
146                }
147            } else if (!seDeseaParar && msPeriodo > 0) {
148                // Avisar si la ejecución fue lenta o muy ajustada, lo que puede generar
149                // retrasos en la simulación
150                mundo.getLogger().log(Level.WARNING, "Se ha tardado tanto tiempo o más en avanzar al siguiente instante de tiempo como el periodo de avance. ¿Está sobrecargado el sistema?");
151            }
152        }
153    }
154}
