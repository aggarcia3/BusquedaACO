003package esei.si.alejandrogg.busquedaACO.mapas;
004
005import java.util.HashMap;
006import java.util.Map;
007import java.util.AbstractMap.SimpleEntry;
008
009import esei.si.alejandrogg.busquedaACO.GrafoCarreteras;
010import esei.si.alejandrogg.busquedaACO.Carretera;
011import esei.si.alejandrogg.busquedaACO.Localidad;
012
013/**
014 * Implementa lógica de uso muy común por parte de grafos de carreteras.
015 * @author Alejandro González García
016 */
017abstract class BaseMapa implements GrafoCarreteras {
018    /**
019     * Relaciona pares de localidades con la carretera que las conecta en un grafo de carreteras.
020     * Es responsabilidad de la subclase la inserción de valores en este mapa, aunque la superclase
021     * se encarga de crearlo en su constructor. La subclase debe de insertar en este mapa dos
022     * entradas por cada carretera, cubriendo ambas permutaciones de localidades que la componen.
023     */
024    protected final Map<SimpleEntry<Localidad, Localidad>, Carretera> carreterasLocalidades;
025
026    /**
027     * Crea un nuevo grafo de carreteras, con un número determinado de carreteras.
028     * Las implementaciones de esta clase deben de llamar a este constructor en el suyo, y acto
029     * seguido rellenar los atributos protegidos de esta clase como corresponda.
030     * @param numeroCarreteras El número de carreteras presentes en el grafo de carreteras.
031     */
032    protected BaseMapa(final int numeroCarreteras) {
033        this.carreterasLocalidades = new HashMap<>((int) ((numeroCarreteras * 2) / 0.75f) + 1, 0.75f);
034    }
035
036    @Override
037    public Carretera getCarretera(final Localidad a, final Localidad b) {
038        return carreterasLocalidades.get(new SimpleEntry<>(a, b));
039    }
040}
