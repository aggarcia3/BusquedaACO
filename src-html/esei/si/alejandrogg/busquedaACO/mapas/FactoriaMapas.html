003package esei.si.alejandrogg.busquedaACO.mapas;
004
005import java.lang.reflect.Constructor;
006import java.lang.reflect.InvocationTargetException;
007import java.util.Map;
008import java.util.HashMap;
009import java.util.NoSuchElementException;
010
011import esei.si.alejandrogg.busquedaACO.GrafoCarreteras;
012
013/**
014 * Se encarga de la creación a demanda de mapas del mundo, que son grafos de carreteras
015 * con datos asociados.
016 * @author Alejandro González García
017 */
018public final class FactoriaMapas {
019    /**
020     * El número de mapas disponibles en el sistema. Es recomendable incrementarlo o decrementarlo
021     * según se añadan o eliminen mapas, aunque no necesario.
022     */
023    private static final short NUM_MAPAS = 1;
024    /**
025     * Asocia un nombre de mapa con su constructor, para agilizar la creación repetida de mapas.
026     */
027    private static final Map<String, Constructor<? extends GrafoCarreteras>> constructores = new HashMap<>((int) (NUM_MAPAS / 0.75f) + 1);
028
029    static {
030        try {
031            // Añadir mapas aquí, copiando y pegando la siguiente línea y modificándola donde corresponda
032            constructores.put(Rumania.NOMBRE, Rumania.class.getDeclaredConstructor());
033        } catch (Throwable t) {
034            // Para este inicializador, tratar checked excepctions (que no heredan de RuntimeException) como
035            // si fuesen unchecked. No tiene mucho sentido manejarlas aquí
036            throw new ExceptionInInitializerError(t);
037        }
038    }
039
040    /**
041     * Crea el grafo de carreteras identificado por el nombre especificado.
042     * @param nombre El nombre identificativo del grafo.
043     * @return El grafo de carreteras que se desea crear.
044     * @throws NoSuchElementException Si el grafo de carreteras especificado no existe, o no se pudo crear.
045     */
046    public static GrafoCarreteras grafoCarreteras(final String nombre) {
047        if (!constructores.containsKey(nombre)) {
048            throw new NoSuchElementException("El mapa " + nombre + " no está disponible en el sistema.");
049        }
050
051        try {
052            return constructores.get(nombre).newInstance();
053        } catch (IllegalAccessException | IllegalArgumentException | InstantiationException | InvocationTargetException exc) {
054            throw new NoSuchElementException("El mapa " + nombre + " existe, pero no se ha podido cargar. Mensaje de error: " + exc.getLocalizedMessage());
055        }
056    }
057}
